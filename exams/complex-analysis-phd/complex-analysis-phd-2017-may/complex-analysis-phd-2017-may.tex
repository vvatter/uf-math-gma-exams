\begin{examinstructions}
Do six of eight. Throughout, \(\mathbb{D}=\{ |z|<1\}\) and \(\mathbb{H}=\{\operatorname{Re} z>0\}\).
\end{examinstructions}
\begin{examproblems}{0}{H2}
\def\ProblemHeadingText{Problem 1.}
\item
Use Liouville’s Theorem to prove the fundamental theorem of algebra.
\def\ProblemHeadingText{Problem 2.}
\item
Suppose \(f:\mathbb{C}\to\mathbb{C}\) is entire and \(|f(z)|=1\) for \(|z|=1\). What can be said about~\(f\)?
\def\ProblemHeadingText{Problem 3.}
\item
Find
\[
\int_0^\pi \frac{dt}{2+\cos(t)}.
\]
\def\ProblemHeadingText{Problem 4.}
\item
Let \(\mathbb{P}=\{0<|z|<1\}\).
\begin{examsubparts}
\item[\textnormal{(a)}]
Does there exist an onto analytic map \(f:\mathbb{D}\to\mathbb{P}\)? If so, can~\(f\) be one-one?
\item[\textnormal{(b)}]
Find a conformal map from the strip \(S=\{0<\operatorname{Im} z<1\}\) to~\(\mathbb{D}\).
\end{examsubparts}
\def\ProblemHeadingText{Problem 5.}
\item
Suppose \(f:\mathbb{C}\to\mathbb{C}\) is entire, non-constant and periodic of period 1, meaning \(f(z+1)=f(z)\). Prove the entire function \(g:\mathbb{C}\to\mathbb{C}\) given by \(g(z)=f(z)-z\) is onto.
\def\ProblemHeadingText{Problem 6.}
\item
Suppose~\(f\) is analytic in an open set containing \(\{|z|\leq 1\}\) and \(|f(z)|<1\) for \(|z|=1\). Prove \(f(z)=z\) has exactly one solution (counting with multiplicity) in~\(\mathbb{D}\).
\def\ProblemHeadingText{Problem 7.}
\item
Let
\[
\mathcal{F}=\{f:\mathbb{D}\to\mathbb{H}: f\text{ is analytic and }f(0)=1\}.
\]
 Prove~\(\mathcal{F}\) is a normal family.
\def\ProblemHeadingText{Problem 8.}
\item
Fix \(0<r<1\) and let \(\mathbb{A}=\{r<|z|<1\}\). Show, if~\(u\) is harmonic on~\(\mathbb{A}\), then there is an analytic function~\(f\) on~\(\mathbb{A}\) and a real constant~\(c\) such that
\[
u(z)=\operatorname{Re} f(z)+c\log(|z|).
\]
\end{examproblems}
