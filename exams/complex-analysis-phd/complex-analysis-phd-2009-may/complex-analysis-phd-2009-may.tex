\begin{examinstructions}
Give complete proofs and computations. Partial credit will be given where justified. In the following, \(\mathbb{C}\) denotes the set of complex numbers and \(D=\{z\in\mathbb{C}\mid |z|<1\}\).
\end{examinstructions}
\begin{examproblems}{0}
\item
Assume that~\(f\) is an entire function with \(f(0)=0\). Consider the family of functions \(\{f_n\}\), where \(f_n\) is the \(n^{\text{th}}\) iterate of~\(f\), i.e. \(f_n=f\circ f\circ\cdots\circ f\) (\(n\) times).
\begin{examsubparts}
\item[\textnormal{(a)}]
If \(|f'(0)|<1\), show there is a open set~\(U\) containing zero so that the family \(\{f_n\}\) is normal when restricted to~\(U\).
\item[\textnormal{(b)}]
If \(|f'(0)|>1\), show there is no open set~\(U\) containing zero so that the family \(\{f_n\}\) is normal when restricted to~\(U\).
\end{examsubparts}
\item
Let~\(G\) and \(G'\) be open, simply connected proper subsets of~\(\mathbb{C}\). Show that for any pair of points \(p\in G\) and \(p'\in G'\) there is a bijective, analytic function \(f:G\to G'\) with \(f(p)=p'\).
\item
This problem has two parts.
\begin{examsubparts}
\item[\textnormal{(a)}]
Show that 
\[
\prod_{n=2}^{\infty}\left(1-\frac{1}{n^2}\right)=\frac{1}{2}.
\]
\item[\textnormal{(b)}]
For which values of~\(z\) does the following product converge? 
\[
\prod_{n=0}^{\infty}(1+z^{3n})
\]
\end{examsubparts}
\item
Find an explicit biholomorphism from the semidisk \(E:=\{z\mid |z|<1,\operatorname{Re}(z)>0\}\) onto the unit disk~\(D\).
\item
Let \(0<a<1\). Evaluate the integral 
\[
\int_0^\infty \frac{x^{a-1}}{1+x}\,dx.
\]
\item
Let~\(f\) be an entire function of finite order. Prove that if the order is not an integer, then~\(f\) must have infinitely many zeros. Does there exist an entire function of infinite order with finitely many zeros? Explain.
\item
Let~\(f\) be entire and assume that there exist \(M,R>0\) and \(n\in\mathbb{N}\) such that \(|f(z)|\leq M|z|^n\) for all \(z\in\mathbb{C}\setminus B(0;R)\). Prove that~\(f\) is a polynomial of degree less than or equal to~\(n\).
\item
Let~\(f\) have a power series expansion about~\(0\) with radius of convergence~\(1\). Assume, in addition, that all of the coefficients of this power series expansion are greater than or equal to zero. Prove that~\(1\) must be a singular point for~\(f\).
\end{examproblems}
