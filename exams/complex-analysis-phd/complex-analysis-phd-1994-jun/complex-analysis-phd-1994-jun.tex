\begin{examproblems}{0}
\item
Show, if \(a>e\), then \(e^z-az^n\) has~\(n\) zeros in \(|z|<1\).
\item
Determine all entire functions~\(f\) such that \(|f|=1\) for \(|z|=1\).
\item
Let~\(G\) denote the punctured disc \(0<|z|<\delta\) and let \(f=z\cos(1/z)\). Determine the set \(f(G)\).
\item
Let~\(f\) be an entire function with a finite number of zeros. Show if for each \(c>0\) 
\[
\lim_{z\to\infty}\frac{|f(z)|}{e^{c|z|}}=0,
\]
 then~\(f\) is a polynomial.
\item
Let \(a_n\geq 0\) and \(f=\sum_{n=0}^{\infty}a_nz^n\) with radius of convergence \(\infty>R>0\). Show that \(z=R\) is a singularity of~\(f\). (Suggestion: Show, if \(R>r>0\) and 
\[
\sum \frac{f^{(n)}(r)}{n!}(z-r)^n
\]
 has radius of convergence \(R'\), then for each \(|z_0|=r\) 
\[
\sum \frac{f^{(n)}(z_0)}{n!}(z-z_0)^n
\]
 has radius of convergence at least \(R'\). Conclude~\(f\) is analytic in \(|z|<R'+r\).)
\item
Let 
\[
f(z)=1+\frac{1}{2^z}+\frac{1}{4^z}+\frac{1}{5^z}+\frac{1}{7^z}+\frac{1}{8^z}+\cdots.
\]
 Show that~\(f\) is analytic in \(\operatorname{Re}z>1\). Analytically continue~\(f\) beyond this half plane. Is~\(f\) analytic at \(z=1\)?
\item
This problem has two parts.
\begin{examsubparts}
\item[\textnormal{a.}]
Show if~\(u\) is harmonic in a region~\(G\) then 
\[
\frac{\partial u}{\partial z}=\frac{\partial u}{\partial x}-i\frac{\partial u}{\partial y}
\]
 is analytic in~\(G\).
\item[\textnormal{b.}]
Show if~\(u\) is harmonic on the annulus \(G=\{z:r<|z|<R\}\), then there exists an analytic function~\(f\) on~\(G\) and a constant~\(c\) such that 
\[
u=\operatorname{Re}f+c\log(|z|).
\]
\end{examsubparts}
\end{examproblems}
