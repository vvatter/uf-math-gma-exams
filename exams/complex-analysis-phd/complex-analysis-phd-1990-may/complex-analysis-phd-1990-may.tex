\begin{examproblems}{0}
\item
Show that \(u(x,y)=x^3-3xy^2+x\) is harmonic in \(\mathbb{R}^2\). Find a harmonic conjugate for~\(u\).
\item
Let~\(G\) be the region which is the intersection of the disks \(|z|<1\) and \(|z-1|<1\). Find a function which maps~\(G\) conformally onto the unit disk.
\item
Evaluate \(\displaystyle \int_0^\infty \frac{\cos x}{(1+x^4)^2}\,dx\) using contour integration.
\item
Show that \(\displaystyle f(s)=\sum_{n=1}^\infty \frac{(-1)^{n-1}}{n^s}\) is analytic in \(\operatorname{Re}s=\sigma>1\). Prove that~\(f\) can be analytically continued to all of~\(\mathbb{C}\) as an entire function. (Hint: Write \(f(s)\) in terms of \(\zeta(s)\) in \(\sigma>1\). Use properties of \(\zeta(s)\) to show that~\(f\) can be made entire.)
\item
Let \(\displaystyle f(z)=\sum_{n=1}^\infty \frac{z^n}{n^2}\). What is the radius of convergence~\(R\) of this series? Is the series convergent at all points \(|z|=R\)? At what points of \(|z|=R\) is~\(f\) analytic? (Hint: Think of~\(f\) as the primitive of \(-\log(1-z)/z\).)
\item
Find all entire functions~\(f\) such that \(|f(z)|=1\) whenever \(|z|=1\).
\item
We say that \(z_0\) is a fixed point of~\(f\) if \(f(z_0)=z_0\). Suppose~\(f\) is analytic in a region~\(D\) containing the closed unit disk.
\begin{examsubparts}
\item[\textnormal{(a)}]
How many fixed points must~\(f\) have in this disk?
\item[\textnormal{(b)}]
If \(|f(z)|>2\) when \(|z|=1\) and \(f(0)=1\), must~\(f\) have a zero in this unit disk? Justify your answers in both parts.
\end{examsubparts}
\item
Suppose~\(f\) is an entire function and \(\operatorname{Re}f(z)\leq A+B|z|^k\) for all~\(z\), where \(A,B\) and~\(k\) are positive numbers. Prove that~\(f\) must be a polynomial. Describe the polynomial as detail as possible when \(A=0\).
\item
Suppose that~\(f\) is a conformal map of unit disk onto a square with center at~\(0\) and \(f(0)=0\). Prove that \(f(iz)=if(z)\). If \(\displaystyle f(z)=\sum^\infty c_nz^n\), prove that \(c_n=0\) unless \(n-1\) is a multiple of~\(4\). (Hint: Show that \(if(z)\) is also a conformal map of the unit disk onto the same square.)
\item
Deduce from the identity 
\[
\sin \pi z=\pi z\prod_{n=1}^\infty\left(1-\frac{z^2}{n^2}\right)
\]
 that 
\[
\sum_{n=1}^\infty (n^2+k^2)^{-1}=\frac{\pi}{2k}\frac{e^{2\pi k}+1}{e^{2\pi k}-1}-\frac{1}{2k^2}.
\]
\end{examproblems}
