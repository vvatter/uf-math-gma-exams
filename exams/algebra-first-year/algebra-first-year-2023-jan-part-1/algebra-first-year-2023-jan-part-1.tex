\begin{examinstructions}
Answer four problems. State clearly all the results from the text that you need, and write your answers clearly in complete English sentences. In all cases, be sure to justify your answer.
\end{examinstructions}
\begin{examproblems}{0}
\item
Denote by \(Q_8\) the quaternion group. Show that if \(f:Q_8\to S_n\) is an injective homomorphism then \(n\geq 8\).
\item
Show that any group of order \(80=2^4\cdot 5\) is solvable (do not simply quote Burnside’s theorem).
\item
This problem has three parts.
\begin{examsubparts}
\item[\textnormal{(i)}]
Define what it means for a group~\(G\) to be nilpotent.
\item[\textnormal{(ii)}]
Show that any nilpotent group is solvable.
\item[\textnormal{(iii)}]
Give an example of a finite solvable group that is not nilpotent, and prove that this example is correct.
\end{examsubparts}
\item
Find
\begin{examsubparts}
\item[\textnormal{(i)}]
representatives of the isomorphism classes of abelian groups of order \(504=2^3\cdot 3^2\cdot 7\), and
\item[\textnormal{(ii)}]
the number of elements of order~\(6\) in each such group.
\end{examsubparts}
\item
This problem has two parts.
\begin{examsubparts}
\item[\textnormal{(i)}]
Find all conjugacy classes of elements of order~\(2\) in \(S_6\), and the number of elements in each class.
\item[\textnormal{(ii)}]
Do the same for \(A_6\).
\end{examsubparts}
\item
Suppose~\(p\) is prime, \(P\subseteq S_p\) is a Sylow~\(p\)-subgroup of \(S_p\) and \(N=N(P)\) is the normalizer of~\(P\). Show that 
\[
N\simeq (\mathbb{Z}/p\mathbb{Z})\rtimes_\varphi(\mathbb{Z}/p\mathbb{Z})^\times
\]
 and describe~\(\varphi\).
\end{examproblems}
