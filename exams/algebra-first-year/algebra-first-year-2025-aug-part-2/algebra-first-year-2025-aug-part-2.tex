\begin{examinstructions}
Answer 4 questions. Write your solutions clearly in complete English sentences. You may quote theorems (within reason) as long as you state them clearly.
\end{examinstructions}
\begin{examproblems}{0}
\item
Let~\(R\) be a commutative ring with \(1 \ne 0\).
\begin{examsubparts}
\item[\textnormal{(a)}]
Define what it means for an ideal of~\(R\) to be maximal.
\item[\textnormal{(b)}]
Prove that every proper ideal of~\(R\) is contained in a maximal ideal.
\end{examsubparts}
\item
This problem has two parts.
\begin{examsubparts}
\item[\textnormal{(a)}]
Define the term Euclidean domain.
\item[\textnormal{(b)}]
Prove that the polynomial ring \(k[x]\), where~\(k\) is a field, is a Euclidean domain.
\end{examsubparts}
\item
This problem has two parts.
\begin{examsubparts}
\item[\textnormal{(a)}]
Give the definition of the term unique factorization domain.
\item[\textnormal{(b)}]
Show that the ring \(\mathbb{Z}[\sqrt{-5}]\) is not a unique factorization domain.
\end{examsubparts}
\item
Let~\(k\) be a field, let \(k[x,y]\) denote the polynomial ring in two variables over~\(k\). Let \(R=k[x,y]/(xy)\) and denote the image in~\(R\) of a polynomial \(f\in k[x,y]\) by \(\bar f\).
\begin{examsubparts}
\item[\textnormal{(a)}]
Prove that if~\(P\) is a prime ideal of~\(R\) then either \(\bar x\in P\) or \(\bar y\in P\) (or both).
\item[\textnormal{(b)}]
Describe all of the prime ideals of~\(R\) that contain \(\bar y\).
\end{examsubparts}
\item
Using the rational canonical form, find one representative for each conjugacy class in \(\operatorname{GL}(5,\mathbb{F}_2)\) whose elements have order~\(4\).
\item
Let~\(L\) be an algebraic field extension of the field~\(K\) and let~\(R\) be a subring of~\(L\) that contains~\(K\). Prove that~\(R\) is a field.
\end{examproblems}
