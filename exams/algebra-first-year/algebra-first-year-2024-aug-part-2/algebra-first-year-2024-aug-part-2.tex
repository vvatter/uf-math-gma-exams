\begin{examinstructions}
Answer four problems. (If you turn in more, the first four will be graded.) Within reason, you may quote theorems as long as you state them clearly.
\end{examinstructions}
\begin{examproblems}{0}
\item
Show that the ring \(\mathbb{Z}[\sqrt{-5}]\) is not a principal ideal domain, and give an example of a non-principal ideal in this ring.
\item
Let \(R=\mathbb{Z}[\sqrt{-1}]\) be the ring of Gaussian integers. Let~\(M\) be a maximal ideal of~\(R\). Prove that \(R/M\) is a finite field of order~\(p\) or \(p^2\) where~\(p\) is a (rational) prime number.
\item
Let~\(V\) be a vector space over a field~\(F\) and let~\(S\) be a subset of~\(V\) such that \(\operatorname{Span}(S)=V\). Prove that there is a subset~\(B\) of~\(S\) which is a basis for~\(V\). (You may not assume that~\(S\) is finite).
\item
Which of the following polynomials are irreducible? Explain.
\begin{examsubparts}
\item[\textnormal{(a)}]
\(X^4+X^2+1\) in \(\mathbb{F}_2[X]\);
\item[\textnormal{(b)}]
\(X^3+X^2+1\) in \(\mathbb{Q}[X]\);
\item[\textnormal{(c)}]
\(X^5+6X^2+4X+18\) in \(\mathbb{Q}[X]\).
\end{examsubparts}
\item
Find representatives of all conjugacy classes of elements of order 5 in \(GL_3(\mathbb{F}_{19})\). Hint: \(x^5-1=(x-1)(x^2-4x+1)(x^2+5x+1)\) in \(\mathbb{F}_{19}[x]\), and the quadratic factors are irreducible.
\item
Let \(E/F\) be a field extension and let~\(R\) be a subring of~\(E\) which contains~\(F\).
\begin{examsubparts}
\item[\textnormal{(a)}]
Prove that if \(E/F\) is an algebraic extension then~\(R\) is a field.
\item[\textnormal{(b)}]
Give an example where \(E/F\) is not an algebraic extension and~\(R\) is not a field.
\end{examsubparts}
\end{examproblems}
