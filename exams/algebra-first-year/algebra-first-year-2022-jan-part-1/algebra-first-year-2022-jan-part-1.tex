\begin{examinstructions}
Answer 4 questions. Write your solutions clearly in complete English sentences. You may quote results (within reason) as long as you state them clearly.
\end{examinstructions}
\begin{examproblems}{0}{H2}
\def\ProblemHeadingText{Problem 1.}
\item
Let~\(G\) be a group of order \(2022=2\cdot 3\cdot 337\) (as a product of primes).
\begin{examsubparts}
\item[\textnormal{(a)}]
Prove that~\(G\) has a normal subgroup of index~\(2\).
\item[\textnormal{(b)}]
Is there a group of order \(2022\) with trivial center?
\end{examsubparts}
\def\ProblemHeadingText{Problem 2.}
\item
Let~\(G\) be a finite group of order \(p^a\), where~\(p\) is a prime and~\(a\) is a positive integer. Prove that if~\(N\) is a nontrivial normal subgroup of~\(G\), then \(N\cap Z(G)\ne\{1\}\).
\def\ProblemHeadingText{Problem 3.}
\item
Prove that if a group~\(G\) has a proper subgroup of finite index, then it has a proper normal subgroup of finite index.
\def\ProblemHeadingText{Problem 4.}
\item
Let~\(G\) be the additive group of rational numbers.
\begin{examsubparts}
\item[\textnormal{(a)}]
Prove that~\(G\) is not cyclic.
\item[\textnormal{(b)}]
Prove that~\(G\) cannot be written as the direct product of two proper subgroups.
\end{examsubparts}
\def\ProblemHeadingText{Problem 5.}
\item
Give the definition of a solvable (finite) group. (You may give any of the equivalent definitions.) Prove, using your definition, that the symmetric group on four letters is solvable.
\def\ProblemHeadingText{Problem 6.}
\item
Let~\(G\) be a finite group and~\(p\) a prime. Let~\(K\) be the subgroup of~\(G\) generated by all elements of order relatively prime to~\(p\).
\begin{examsubparts}
\item[\textnormal{(a)}]
Prove that~\(K\) is a characteristic subgroup of~\(G\).
\item[\textnormal{(b)}]
Prove that \(|G/K|\) is a power of~\(p\).
\end{examsubparts}
\end{examproblems}
