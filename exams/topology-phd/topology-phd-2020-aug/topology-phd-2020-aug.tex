\begin{examinstructions}
Work the following problems and show all work. Support all statements to the best of your ability.
\end{examinstructions}
\begin{examproblems}{0}{H2}
\def\ProblemHeadingText{Problem 1.}
\item
Let~\(X\) be a topological space. Given two points \(a,b\in X\), a collection of sets \(A_1,\ldots,A_n\) is a simple chain from~\(a\) to~\(b\) if \(a\in A_1\), \(b\in A_n\), and \(A_i\cap A_j\ne\varnothing\) if and only if \(|i-j|\le 1\). Prove that if \(\{U_\alpha\}\) is a collection of open sets covering~\(X\) and~\(X\) is connected, then there is a simple chain of elements in \(\{U_\alpha\}\) joining any pair of points \(a,b\in X\). (Hint: Consider the set \(C_a\) of all~\(b\) so that there is a simple chain of elements joining~\(a\) and~\(b\). Show that \(C_a\) is both open and closed.)
\def\ProblemHeadingText{Problem 2.}
\item
Let \(X=S^2\cup_f e^3\cup_h e^5\), where \(f:S^2\to S^2\) has degree~\(n\) and \(h:S^4\to S^2\cup_f e^3\) is continuous.
\begin{examsubparts}
\item[\textnormal{(a)}]
What is the Euler characteristic of~\(X\)?
\item[\textnormal{(b)}]
Determine \(H_\bullet(X;\mathbb Z)\) and \(H^\bullet(X;\mathbb Z_2)\) as a function of~\(n\).
\item[\textnormal{(c)}]
Can~\(X\) have the homotopy type of a closed~\(5\)-manifold? (If the answer depends on~\(n\), explain.)
\end{examsubparts}
\def\ProblemHeadingText{Problem 3.}
\item
Let~\(X\) be the space \(\mathbb{RP}^3/\mathbb{RP}^1\) (the space obtained by collapsing \(\mathbb{RP}^1\subset\mathbb{RP}^3\) to a point). Compute the integral homology groups of~\(X\).
\def\ProblemHeadingText{Problem 4.}
\item
Let \(\mathbb{RP}^2\) be the real projective plane and fix a basepoint \(x_0\in\mathbb{RP}^2\). Note that we can realize the wedge sum \(\mathbb{RP}^2\vee\mathbb{RP}^2\) as \((\mathbb{RP}^2\times\{x_0\})\cup(\{x_0\}\times\mathbb{RP}^2)\subset\mathbb{RP}^2\times\mathbb{RP}^2\).
\begin{examsubparts}
\item[\textnormal{(a)}]
Compute the fundamental groups of \(\mathbb{RP}^2\vee\mathbb{RP}^2\) and \(\mathbb{RP}^2\times\mathbb{RP}^2\).
\item[\textnormal{(b)}]
Show that \(\mathbb{RP}^2\vee\mathbb{RP}^2\) is not a retract of \(\mathbb{RP}^2\times\mathbb{RP}^2\).
\item[\textnormal{(c)}]
Show that any map \(f:\mathbb{RP}^2\vee\mathbb{RP}^2\to S^1\) is null-homotopic.
\end{examsubparts}
\def\ProblemHeadingText{Problem 5.}
\item
Compute \(H_\bullet(M;\mathbb Z_2)\), where~\(M\) is the space with integral homology groups
\[
H_k(M;\mathbb Z)=
\begin{cases}
\mathbb Z & k=0,2,4,\\
\mathbb Z_2 & k=1,\\
\mathbb Z_3 & k=3,\\
0 & \text{otherwise}.
\end{cases}
\]
\end{examproblems}
\begin{examinstructions}
Answer the following with complete definitions, statements, or short proofs.
\end{examinstructions}
\begin{examproblems}{5}{H2}
\def\ProblemHeadingText{Problem 6.}
\item
Let~\(M\) be an orientable simply-connected~\(3\)-manifold. Compute the integral homology of~\(M\).
\def\ProblemHeadingText{Problem 7.}
\item
Compute \(\chi(\mathbb{RP}^2\times\mathbb{CP}^4\times S^4)\).
\def\ProblemHeadingText{Problem 8.}
\item
Give an example of a space that is connected but not locally connected.
\def\ProblemHeadingText{Problem 9.}
\item
State the Urysohn Lemma.
\def\ProblemHeadingText{Problem 10.}
\item
Prove that if \(m\ne n\), then \(\mathbb R^m\) is not homeomorphic to \(\mathbb R^n\).
\def\ProblemHeadingText{Problem 11.}
\item
Let \(\mathbb N\) be the set of natural numbers and let \(\mathbb N^*\) be its one-point compactification. Let~\(X\) be any topological space. Prove that a sequence \(\{a_n\}\) in~\(X\) converges to~\(a\) if and only if the map \(f:\mathbb N^*\to X\) defined by \(f(n)=a_n\) and \(f(\infty)=a\) is continuous. (Here \(\mathbb N\) has the discrete topology.)
\def\ProblemHeadingText{Problem 12.}
\item
State the Borsuk–Ulam Theorem.
\def\ProblemHeadingText{Problem 13.}
\item
Describe the possible connected covering spaces of \(S^1\times S^1\).
\def\ProblemHeadingText{Problem 14.}
\item
Does the following exact sequence of abelian groups necessarily split? Prove or give a counterexample.
\[
0\to\mathbb Z\to\mathbb Z\oplus\mathbb Z\to A\to 0
\]
\def\ProblemHeadingText{Problem 15.}
\item
Compute the integral homology of the space \(\mathbb{CP}^2\times\mathbb{RP}^3\).
\end{examproblems}
