\begin{examinstructions}
Work the following problems and show all work. Support all statements to the best of your ability.
\end{examinstructions}
\begin{examproblems}{0}{H2}
\def\ProblemHeadingText{Problem 1.}
\item
Show that there is no map \(S^2 \to T^2\) of degree 2. Does there exist a map \(f:T^2 \to S^2\) of degree 2?
\def\ProblemHeadingText{Problem 2.}
\item
Does every function \(f(x)=x^2\) admit a continuous extension \(\bar f:\beta\mathbb{R}\to\mathbb{R}\) to the Stone–Čech compactification?
\def\ProblemHeadingText{Problem 3.}
\item
Is the 2-sphere \(S^2\) with~\(k\) points removed homeomorphic to a topological group for
\begin{examsubparts}
\item[\textnormal{(a)}]
\(k=0\)?
\item[\textnormal{(b)}]
\(k=1\)?
\item[\textnormal{(c)}]
\(k=2\)?
\item[\textnormal{(d)}]
\(k=3\)?
\end{examsubparts}
\def\ProblemHeadingText{Problem 4.}
\item
Prove that the topologist sin curve is not path connected.
\def\ProblemHeadingText{Problem 5.}
\item
Suppose that \(X=C_1\cup C_2\) where each \(C_i\) is homeomorphic to the topologist sin curve. Can~\(X\) be path connected?
\end{examproblems}
\begin{examinstructions}
Answer the following with complete definitions or statements or short proofs.
\end{examinstructions}
\begin{examproblems}{5}{H2}
\def\ProblemHeadingText{Problem 6.}
\item
Why the~\(n\)-sphere \(S^n\) is not homeomorphic to the~\(n\)-space \(\mathbb{R}^n\)?
\def\ProblemHeadingText{Problem 7.}
\item
State the Contraction Mapping Theorem.
\def\ProblemHeadingText{Problem 8.}
\item
Compute the Euler characteristic \(\chi(\mathbb{C}P^5\times\mathbb{R}P^5\times S^5\times S^2)\).
\def\ProblemHeadingText{Problem 9.}
\item
State the Baire Category Theorem.
\def\ProblemHeadingText{Problem 10.}
\item
Does a surface of genus 2 admit a nonzero vector field?
\def\ProblemHeadingText{Problem 11.}
\item
State the Five Lemma.
\def\ProblemHeadingText{Problem 12.}
\item
What can you say about the~\(k\)-th cohomology group of a closed oriented~\(n\)-manifold for
\begin{examsubparts}
\item[\textnormal{(a)}]
\(k=n\)?
\item[\textnormal{(b)}]
\(k=n-1\)?
\end{examsubparts}
\def\ProblemHeadingText{Problem 13.}
\item
Does there exist a covering space of the figure eight that has a non-trivial abelian fundamental group?
\def\ProblemHeadingText{Problem 14.}
\item
State the Künneth Formula for homology.
\def\ProblemHeadingText{Problem 15.}
\item
State the Lefschetz Fixed Point Theorem.
\end{examproblems}
