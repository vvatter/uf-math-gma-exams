\begin{examinstructions}
Work the following problems and show all work. Support all statements to the best of your ability.
\end{examinstructions}
\begin{examproblems}{0}{H2}
\def\ProblemHeadingText{Problem 1.}
\item
Compute \(\pi_1(S^2)\) and \(H_*(S^2)\).
\def\ProblemHeadingText{Problem 2.}
\item
Let~\(X\) be a connected completely regular topological space having more than one point. Can~\(X\) be countable?
\def\ProblemHeadingText{Problem 3.}
\item
Does there exist a covering space of the figure eight with nontrivial abelian fundamental group?
\def\ProblemHeadingText{Problem 4.}
\item
Show that the 2-dimensional sphere with four points deleted cannot be a topological group.
\def\ProblemHeadingText{Problem 5.}
\item
Prove that there is no map of degree two from \(S^2\) to the torus \(T^2\).
\end{examproblems}
\begin{examinstructions}
Answer the following with complete definitions or statements or short proofs.
\end{examinstructions}
\begin{examproblems}{5}{H2}
\def\ProblemHeadingText{Problem 6.}
\item
State the Baire Category Theorem.
\def\ProblemHeadingText{Problem 7.}
\item
State the homology Mayer–Vietoris Theorem.
\def\ProblemHeadingText{Problem 8.}
\item
State the Urysohn Lemma.
\def\ProblemHeadingText{Problem 9.}
\item
State the Five Lemma.
\def\ProblemHeadingText{Problem 10.}
\item
Compute the Euler characteristic \(\chi(\mathbb{RP}^4 \times S^3 \times T^2)\).
\def\ProblemHeadingText{Problem 11.}
\item
Define retraction and deformation retraction.
\def\ProblemHeadingText{Problem 12.}
\item
State the Lefschetz Fixed Point Theorem.
\def\ProblemHeadingText{Problem 13.}
\item
Can the set of irrational numbers be presented as a countable union of closed in~\(\mathbb{R}\) subsets?
\def\ProblemHeadingText{Problem 14.}
\item
Draw a picture of the universal cover of the 2-sphere with the segment joining the north and south poles.
\def\ProblemHeadingText{Problem 15.}
\item
Describe all connected subsets of~\(\mathbb{R}\).
\end{examproblems}
