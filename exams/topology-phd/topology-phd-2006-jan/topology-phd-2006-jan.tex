\begin{examinstructions}
Work the following problems and show all work. Support all statements to the best of your ability.
\end{examinstructions}
\begin{examproblems}{0}{H2}
\def\ProblemHeadingText{Problem 1.}
\item
Describe all metric spaces whose completions are compact.
\def\ProblemHeadingText{Problem 2.}
\item
Describe (with a proof) all connected subsets of the real line~\(\mathbb{R}\).
\def\ProblemHeadingText{Problem 3.}
\item
Let~\(X\) be a connected normal space having more than one point. Can~\(X\) be countable?
\def\ProblemHeadingText{Problem 4.}
\item
Show that the 2-torus with a deleted point \(T\setminus\{x_0\}\) is not a retract of~\(T\).
\def\ProblemHeadingText{Problem 5.}
\item
Prove that \(\mathbb{R}^n\) is not homeomorphic to \(\mathbb{R}^m\) for \(n\ne m\).
\end{examproblems}
\begin{examinstructions}
Answer the following with complete definitions or statements or short proofs.
\end{examinstructions}
\begin{examproblems}{5}{H2}
\def\ProblemHeadingText{Problem 6.}
\item
Define the degree of a map. Prove that, given a closed oriented~\(n\)-dimensional manifold~\(M\) and an integer number~\(d\), there exists a map \(f:M\to S^n\) of degree~\(d\).
\def\ProblemHeadingText{Problem 7.}
\item
Find a degree of the map \(f:T\to T\), \(f(x)=x^k\), where \(T=S^1\times S^1\) is the 2-dimensional torus and \(x^k=x\cdots x\) is the kth power in the group \(S^1\times S^1\): \(x^k=(z_1^k,z_2^k)\) for \(x=(z_1,z_2)\), \(z_1,z_2\in S^1\subset\mathbb{C}\).
\def\ProblemHeadingText{Problem 8.}
\item
State the Seifert–van Kampen Theorem.
\def\ProblemHeadingText{Problem 9.}
\item
Show that two closed orientable surfaces of different genera are not homeomorphic.
\def\ProblemHeadingText{Problem 10.}
\item
State the Universal Coefficient Theorem relating integral homology with integral cohomology.
\def\ProblemHeadingText{Problem 11.}
\item
Find the fundamental group and homology groups of the Klein bottle.
\def\ProblemHeadingText{Problem 12.}
\item
Formulate the Theorem on Invariance of Domain.
\def\ProblemHeadingText{Problem 13.}
\item
Let~\(X\) be a compact Hausdorff topological space. Describe all compact subsets of the space \(C(X,\mathbb{R})\) of continuous functions \(X\to\mathbb{R}\) in uniform topology.
\def\ProblemHeadingText{Problem 14.}
\item
Define (or describe) the Stone–Čech compactification \(\beta X\) of a topological space~\(X\). Show that \(\beta\mathbb{Z}\) is not metrizable (here~\(\mathbb{Z}\) denotes the set of integer numbers in discrete topology).
\def\ProblemHeadingText{Problem 15.}
\item
Give a definition of CW-space. State the Cellular Approximation Theorem.
\end{examproblems}
