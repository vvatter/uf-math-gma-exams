\begin{examinstructions}
Work the following problems and show all work. Support all statements to the best of your ability.
\end{examinstructions}
\begin{examproblems}{0}{H2}
\def\ProblemHeadingText{Problem 1.}
\item
Is the quotient space \(X=\mathbb{R}/\mathbb{Z}\) obtained from the reals~\(\mathbb{R}\) by collapsing the integers to a point metrizable? Compute homology groups of~\(X\).
\def\ProblemHeadingText{Problem 2.}
\item
Show that any continuous map \(f:\mathbb{RP}^3\to T^3\) is null-homotopic.
\def\ProblemHeadingText{Problem 3.}
\item
Let \(A\subset\mathbb{R}^3\) be the union of a countable family of lines in \(\mathbb{R}^3\). Prove that \(\mathbb{R}^3\setminus A\) is connected.
\def\ProblemHeadingText{Problem 4.}
\item
Does there exist a closed orientable~\(n\)-manifold~\(M\) with
\begin{examsubparts}
\item[\textnormal{(a)}]
\(n=4\) and \(H_2(M)=\mathbb{Z}\)?
\item[\textnormal{(b)}]
\(n=5\) and \(H_3(M)=\mathbb{Z}\)?
\item[\textnormal{(c)}]
\(n=6\) and \(H_3(M)=\mathbb{Z}\)?
\end{examsubparts}
\def\ProblemHeadingText{Problem 5.}
\item
Let \(p:X\to M_2\) be a covering map of an orientable surface of genus 2 that corresponds to the subgroup \(G\subset\pi_1(M_2)\) generated by \(\{a_1,b_1\}\) in \(\pi_1(M_2)=\langle a_1,a_2,b_1,b_2\mid [a_1,b_1][a_2,b_2]=1\rangle\). Compute homology groups of~\(X\).
\end{examproblems}
\begin{examinstructions}
Answer the following with complete definitions or statements or short proofs.
\end{examinstructions}
\begin{examproblems}{5}{H2}
\def\ProblemHeadingText{Problem 6.}
\item
Compute the Euler characteristic of the manifold \(M=S^2\times S^4\times\mathbb{CP}^{51}\times\mathbb{RP}^{99}\).
\def\ProblemHeadingText{Problem 7.}
\item
State the Lefschetz Fixed Point Theorem. Derive the Brouwer Fixed Point Theorem from the Lefschetz Fixed Point Theorem.
\def\ProblemHeadingText{Problem 8.}
\item
Give definitions of retraction, deformation retraction, retract, and deformation retract. Give an example of a retract~\(A\) of a topological space which is not a deformation retract.
\def\ProblemHeadingText{Problem 9.}
\item
Suppose that~\(A\) is a retract of~\(X\). Is it true that \(H_i(A)\) is a direct summand of \(H_i(X)\)? What about \(H^i(A)\) and \(H^i(X)\)?
\def\ProblemHeadingText{Problem 10.}
\item
Give definition of the mapping cylinder. Is the Möbius band a mapping cylinder?
\def\ProblemHeadingText{Problem 11.}
\item
Formulate the Urysohn Lemma.
\def\ProblemHeadingText{Problem 12.}
\item
Give definition of the mapping cone. Is there a map \(f:X\to Y\) with the mapping cone \(C_f\) homeomorphic to \(\mathbb{RP}^3\)?
\def\ProblemHeadingText{Problem 13.}
\item
Which of the following spaces are topological groups?
\begin{examsubparts}
\item[\textnormal{(a)}]
the torus \(T^2\);
\item[\textnormal{(b)}]
the Klein bottle~\(K\);
\item[\textnormal{(c)}]
the projective space \(\mathbb{RP}^3\).
\end{examsubparts}
\def\ProblemHeadingText{Problem 14.}
\item
Is \(\mathbb{R}^{\omega}\) connected in the uniform topology?
\def\ProblemHeadingText{Problem 15.}
\item
State the Five Lemma.
\end{examproblems}
