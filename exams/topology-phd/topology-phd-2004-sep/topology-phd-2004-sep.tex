\begin{examinstructions}
Work the following problems and show all work. Support all statements to the best of your ability.
\end{examinstructions}
\begin{examproblems}{0}{H2}
\def\ProblemHeadingText{Problem 1.}
\item
Show that \(\pi_1(S^2)\) is trivial.
\def\ProblemHeadingText{Problem 2.}
\item
Show that there is no retraction of the Möbius band onto its boundary.
\def\ProblemHeadingText{Problem 3.}
\item
Does there exists a covering space of the figure eight with nontrivial abelian fundamental group?
\def\ProblemHeadingText{Problem 4.}
\item
Show that the 2-dimensional sphere with three points deleted cannot be a topological group.
\def\ProblemHeadingText{Problem 5.}
\item
Let~\(M\) be a 4-dimensional closed simply connected manifold. Show that every continuous map \(f:M\to M\) which is homotopic to the identity has a fixed point.
\end{examproblems}
\begin{examinstructions}
Answer the following with complete definitions or statements or short proofs.
\end{examinstructions}
\begin{examproblems}{5}{H2}
\def\ProblemHeadingText{Problem 6.}
\item
State the Baire Category Theorem.
\def\ProblemHeadingText{Problem 7.}
\item
State the Lefschetz Fixed Point Theorem.
\def\ProblemHeadingText{Problem 8.}
\item
State the Jordan Curve Theorem.
\def\ProblemHeadingText{Problem 9.}
\item
State the Five Lemma.
\def\ProblemHeadingText{Problem 10.}
\item
Show that no covering of the 2-torus has the homotopy type of a figure eight.
\def\ProblemHeadingText{Problem 11.}
\item
Let \(S_r(x,y)\) denote a circle of radius~\(r\) centered at \((x,y)\in\mathbb{R}^2\). Let \(\Sigma R\) be the suspension over Hawaiian ring \(R=\bigcup_{n=1}^{\infty}S_{1/n}(1/n,0)\). Is \(\Sigma R\)
\begin{examsubparts}
\item[\textnormal{(a)}]
locally path connected?
\item[\textnormal{(b)}]
locally simply connected?
\item[\textnormal{(c)}]
semi-locally simply connected?
\end{examsubparts}
\def\ProblemHeadingText{Problem 12.}
\item
State the Tietze Extension Theorem.
\def\ProblemHeadingText{Problem 13.}
\item
Define retraction and deformation retraction.
\def\ProblemHeadingText{Problem 14.}
\item
State the Universal Coefficient formula for homology.
\def\ProblemHeadingText{Problem 15.}
\item
Let~\(X\) be a complete metric space such that no point is isolated. Can~\(X\) be countable?
\end{examproblems}
