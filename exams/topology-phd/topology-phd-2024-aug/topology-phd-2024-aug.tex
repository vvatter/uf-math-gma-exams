\begin{examinstructions}
Work the following problems and show all work. Support all statements to the best of your ability.
\end{examinstructions}
\begin{examproblems}{0}{H2}
\def\ProblemHeadingText{Problem 1.}
\item
Let \(X=\mathbb{R}/\mathbb{Z}\) be the quotient space obtained from the real line~\(\mathbb{R}\) by collapsing the set of integers to a point.
\begin{examsubparts}
\item[\textnormal{(a)}]
Is~\(X\) metrizable?
\item[\textnormal{(b)}]
Does~\(X\) admit a CW complex structure?
\item[\textnormal{(c)}]
Compute the fundamental group of~\(X\).
\item[\textnormal{(d)}]
Compute homology groups of~\(X\).
\end{examsubparts}
\def\ProblemHeadingText{Problem 2.}
\item
This problem has three parts.
\begin{examsubparts}
\item[\textnormal{(a)}]
Prove that the 3-dimensional real projective space \(\mathbb{R}P^3\) is orientable.
\item[\textnormal{(b)}]
Does there exist a degree 2 continuous map \(f:\mathbb{R}P^3\to T^3\) where \(T^3=S^1\times S^1\times S^1\) is a 3-dimensional torus?
\item[\textnormal{(c)}]
Does there exist a degree 2 continuous map \(f:T^3\to\mathbb{R}P^3\)?
\end{examsubparts}
\def\ProblemHeadingText{Problem 3.}
\item
Let \(A\subset\mathbb{R}^2\) be an infinite countable subspace.
\begin{examsubparts}
\item[\textnormal{(a)}]
Can~\(A\) be connected?
\item[\textnormal{(b)}]
Is \(\mathbb{R}^2\setminus A\) connected?
\end{examsubparts}
\def\ProblemHeadingText{Problem 4.}
\item
Does there exist a closed orientable~\(n\)-manifold~\(M\) with
\begin{examsubparts}
\item[\textnormal{(a)}]
\(n=4\) and \(H_2(M)=\mathbb{Z}\)?
\item[\textnormal{(b)}]
\(n=5\) and \(H_3(M)=\mathbb{Z}\)?
\item[\textnormal{(c)}]
\(n=6\) and \(H_3(M)=\mathbb{Z}\)?
\end{examsubparts}
\def\ProblemHeadingText{Problem 5.}
\item
Prove that the fundamental group of a topological group is abelian.
\end{examproblems}
\begin{examinstructions}
Answer the following with complete definitions or statements or short proofs.
\end{examinstructions}
\begin{examproblems}{5}{H2}
\def\ProblemHeadingText{Problem 6.}
\item
Compute the Euler characteristic of the manifold \(M=\mathbb{H}P^3\times\mathbb{C}P^4\times\mathbb{R}P^5\).
\def\ProblemHeadingText{Problem 7.}
\item
State the Lefschetz Fixed Point Theorem. Derive the Brouwer Fixed Point Theorem from the Lefschetz Fixed Point Theorem.
\def\ProblemHeadingText{Problem 8.}
\item
State the Baire Category Theorem.
\def\ProblemHeadingText{Problem 9.}
\item
Suppose that~\(A\) is a retract of~\(X\). Show that \(H_i(A)\) is a direct summand of \(H_i(X)\).
\def\ProblemHeadingText{Problem 10.}
\item
Give the definition of the mapping cylinder. Is the Möbius band a mapping cylinder?
\def\ProblemHeadingText{Problem 11.}
\item
Is the 2-sphere \(S^2\) with~\(k\) points removed homeomorphic to a topological group for
\begin{examsubparts}
\item[\textnormal{(a)}]
\(k=1\)?
\item[\textnormal{(b)}]
\(k=2\)?
\item[\textnormal{(c)}]
\(k=3\)?
\end{examsubparts}
\def\ProblemHeadingText{Problem 12.}
\item
Give the definition of the mapping cone. Is there a map \(f:X\to Y\) with the mapping cone \(C_f\) homeomorphic to \(\mathbb{R}P^3\)?
\def\ProblemHeadingText{Problem 13.}
\item
Which of the following spaces are topological groups?
\begin{examsubparts}
\item[\textnormal{(a)}]
the torus \(T^2\);
\item[\textnormal{(b)}]
the Klein bottle~\(K\);
\item[\textnormal{(c)}]
the projective space \(\mathbb{R}P^3\).
\end{examsubparts}
\def\ProblemHeadingText{Problem 14.}
\item
Is \(\mathbb{R}^{\omega}\) connected in the uniform topology?
\def\ProblemHeadingText{Problem 15.}
\item
State the Five Lemma.
\end{examproblems}
