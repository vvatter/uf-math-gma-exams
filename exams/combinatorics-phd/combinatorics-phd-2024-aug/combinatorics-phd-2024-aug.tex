\begin{examproblems}{0}{H2}
\def\ProblemHeadingText{Problem 1.}
\item
Let \(h(n)\) be the number of words of length~\(n\) over the alphabet \(\{a,b,c\}\) in which an~\(a\) is never immediately followed by a~\(b\). Find the ordinary generating function of the sequence \(\{h(n)\}\).
\def\ProblemHeadingText{Problem 2.}
\item
Let \(H(n)\) be the number of permutations of length~\(n\) that have an even number of fixed points. Find the exponential generating function of the sequence \(\{H(n)\}\).
\def\ProblemHeadingText{Problem 3.}
\item
Select a \(132\)-avoiding permutation~\(\pi\) of length~\(n\) uniformly at random. Which is more likely, that~\(n\) is at either end of~\(\pi\), or that~\(n\) is somewhere between positions~\(2\) and \(n-1\) (inclusive) in~\(\pi\)? Justify your answer.
\def\ProblemHeadingText{Problem 4.}
\item
Let~\(T\) be a tree. Is it true that there is a vertex~\(v\) in~\(T\) that is part of all paths of maximum length in~\(T\)? Justify your answer.
\def\ProblemHeadingText{Problem 5.}
\item
Prove that a graph in which all of the vertices have even degrees contains no bridges. (Recall that an edge~\(e\) of a graph~\(G\) is a bridge if \(G-e\) has more connected components than~\(G\).)
\def\ProblemHeadingText{Problem 6.}
\item
Prove that every convex polyhedron has two faces that have the same number of vertices.
\def\ProblemHeadingText{Problem 7.}
\item
Find the number of all 2-element antichains in the Boolean algebra \(B_n\).
\def\ProblemHeadingText{Problem 8.}
\item
Let~\(P\) be a finite poset. Prove that the number of elements in a maximum chain in~\(P\) equals the number of antichains in the smallest antichain cover of~\(P\). (Recall that an antichain cover of~\(P\) is a set of antichains whose union is~\(P\).)
\end{examproblems}
