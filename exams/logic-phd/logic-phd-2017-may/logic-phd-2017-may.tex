\begin{examinstructions}
Answer six questions, at least one in each section.
\end{examinstructions}
\section{Section 1}
\begin{examproblems}{0}
\item
State and prove Tarski’s criterion for elementarity of submodels.
\item
State and prove the downward Löwenheim–Skolem theorem.
\item
Prove that the models \(\langle \mathbb{Z},\leq\rangle\) (integers with the usual ordering) and \(\langle \mathbb{Z}+\mathbb{Z},\leq\rangle\) (two copies of integers with the usual ordering, one following the other) are elementarily equivalent.
\end{examproblems}
\section{Section 2}
\begin{examproblems}{0}
\item
Prove that for any cardinal~\(\kappa\), the cofinality \(\operatorname{cf}(2^\kappa)>\kappa\).
\item
Show that the wellordering principle implies the Zorn’s lemma.
\item
Show that a c.c.c. forcing preserves all cardinals.
\end{examproblems}
\section{Section 3}
\begin{examproblems}{0}
\item
Show that every infinite computably enumerable set has an infinite computable subset.
\item
Show that there are two subsets of~\(\omega\) which are incomparable in the sense of the Turing ordering.
\item
Explain what a \(\Pi^1_1\) complete set is and provide an example with a proof.
\end{examproblems}
