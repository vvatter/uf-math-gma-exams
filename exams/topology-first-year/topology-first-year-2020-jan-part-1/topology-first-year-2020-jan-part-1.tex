\begin{examinstructions}
Work the following problems and show all work. Support all statements to the best of your ability.
\end{examinstructions}
\begin{examproblems}{0}{H2}
\def\ProblemHeadingText{Problem 1.}
\item
Show that the set of functions \(f:\mathbb Z\to\mathbb Z\) that are eventually~\(0\) is countable.
\def\ProblemHeadingText{Problem 2.}
\item
Let \(A\subset\mathbb R^2\) be a countable subset. Show that \(\mathbb R^2\setminus A\) is connected.
\def\ProblemHeadingText{Problem 3.}
\item
Show that every continuous map \(f:[0,1]\to[0,1]\) has a fixed point. Is this true for continuous maps
\begin{examsubparts}
\item[\textnormal{(a)}]
\(f:[0,1]\to(0,1)\)?
\item[\textnormal{(b)}]
\(f:(0,1)\to(0,1)\)?
\end{examsubparts}
\def\ProblemHeadingText{Problem 4.}
\item
Let~\(X\) be connected, show that
\begin{examsubparts}
\item[\textnormal{(a)}]
\(X^n\) is connected;
\item[\textnormal{(b)}]
\(X^\omega\) is connected.
\end{examsubparts}
\def\ProblemHeadingText{Problem 5.}
\item
Let \(f:X\to X\) be a map of a compact metric space to itself that satisfies the following condition: \(d(f(x),f(y))<d(x,y)\) for \(x\ne y\).
\begin{examsubparts}
\item[\textnormal{(a)}]
Prove that~\(f\) is continuous.
\item[\textnormal{(b)}]
Show that~\(f\) has a fixed point and the fixed point is unique.
\end{examsubparts}
\end{examproblems}
\begin{examinstructions}
Answer the following with complete definitions or statements or short proofs.
\end{examinstructions}
\begin{examproblems}{5}{H2}
\def\ProblemHeadingText{Problem 6.}
\item
State the Intermediate Value Theorem.
\def\ProblemHeadingText{Problem 7.}
\item
Does there exist a continuous surjective map from the 2-sphere \(S^2\) to the interval \((0,1)\)?
\def\ProblemHeadingText{Problem 8.}
\item
Does there exist a continuous injective map from the 2-sphere \(S^2\) to the interval \((0,1)\)?
\def\ProblemHeadingText{Problem 9.}
\item
State the Cantor–Schröder–Bernstein Theorem.
\def\ProblemHeadingText{Problem 10.}
\item
Is every connected space path connected?
\def\ProblemHeadingText{Problem 11.}
\item
What is a basis of a topology? Does the set of all half-open intervals \(\{(a,b],[c,d)\mid a<b,\ c<d\}\) form a basis of a topology on \(\mathbb R\)?
\def\ProblemHeadingText{Problem 12.}
\item
State the Extreme Value Theorem.
\def\ProblemHeadingText{Problem 13.}
\item
Is \(\mathbb R^\omega\) connected in the uniform topology?
\def\ProblemHeadingText{Problem 14.}
\item
Is the set of integers \(\mathbb Z\) locally compact? Is the set of rational numbers \(\mathbb Q\) locally compact?
\def\ProblemHeadingText{Problem 15.}
\item
Is \(\mathbb R^2\) with the dictionary order topology metrizable?
\end{examproblems}
