\subsection{Part 1.}
\begin{examinstructions}
Work each of the following problems and show all work. Support all statements to the best of your ability.
\end{examinstructions}
\begin{examproblems}{0}{H3}
\def\ProblemHeadingText{Problem 1.}
\item
Let~\(S\) be a set, and let \(\mathcal{P}(S)\) denote the set of subsets of~\(S\). Prove that there is no surjective function \(f:S\to\mathcal{P}(S)\).
\def\ProblemHeadingText{Problem 2.}
\item
Let~\(X\) and~\(Y\) be connected topological spaces. Prove that \(X\times Y\) is connected, where \(X\times Y\) has the product topology.
\def\ProblemHeadingText{Problem 3.}
\item
Prove that every compact, metrizable space has a countable basis.
\def\ProblemHeadingText{Problem 4.}
\item
Prove that the one-point compactification of \(\mathbb{Z}_+\) is homeomorphic to the subspace \(\{0\}\cup\left\{\frac{1}{n}:n\in\mathbb{Z}_+\right\}\) of~\(\mathbb{R}\).
\def\ProblemHeadingText{Problem 5.}
\item
Recall that \(\mathbb{R}_K\) denotes the real line with the~\(K\)-topology, and the set~\(K\) is given by \(K=\left\{\frac{1}{n}:n\in\mathbb{Z}_+\right\}\). Let~\(Y\) denote the quotient space obtained from \(\mathbb{R}_K\) by collapsing the set~\(K\) to a point. Prove that~\(Y\) satisfies the \(T_1\) axiom, but is not Hausdorff.
\end{examproblems}
\subsection{Part 2.}
\begin{examinstructions}
Answer the following with complete definitions or statements or short proofs.
\end{examinstructions}
\begin{examproblems}{5}{H3}
\def\ProblemHeadingText{Problem 6.}
\item
In the finite complement topology on~\(\mathbb{R}\), to what point or points does the sequence \(x_n=\frac{1}{n}\) converge?
\def\ProblemHeadingText{Problem 7.}
\item
Is every simply ordered set a Hausdorff space in the order topology?
\def\ProblemHeadingText{Problem 8.}
\item
Define compact, limit point compact, and sequentially compact.
\def\ProblemHeadingText{Problem 9.}
\item
If \(f:X\to Y\) is a continuous surjective function, and~\(X\) is path connected, must~\(Y\) be path connected?
\def\ProblemHeadingText{Problem 10.}
\item
State the Intermediate Value Theorem.
\end{examproblems}
