\begin{examinstructions}
Work the following problems and show all work. Support all statements to the best of your ability.
\end{examinstructions}
\begin{examproblems}{0}
\item
Show that \(S^2\) is not homeomorphic to \(S^4\).
\item
Show that the Stone–Čech compactification \(\beta X\) of~\(X\) is connected if and only if~\(X\) is connected.
\item
Let \(X_{\geq 0}\), \(Y_{\geq 0}\), and \(Z_{\geq 0}\) denote the nonnegative parts of the~\(x\)-axis, the~\(y\)-axis, and the~\(z\)-axis in \(\mathbb{R}^3\) respectively. Let \(A=\mathbb{R}^3\setminus(X_{\geq 0}\cup Y_{\geq 0}\cup Z_{\geq 0})\). Is the fundamental group \(\pi_1(A)\) abelian?
\item
Show that the torus~\(T\) is not homeomorphic to the Klein bottle~\(K\).
\item
By the definition the triod (or the letter T space) is a space homeomorphic to \(X=([0,1]\times\{0\})\cup(\{0\}\times[-1,1])\subset\mathbb{R}^2\). Show that every continuous map of the triod to itself has a fixed point.
\end{examproblems}
\begin{examinstructions}
Answer the following with complete definitions or statements or proofs.
\end{examinstructions}
\begin{examproblems}{5}
\item
Does the sequence \(f_n:\mathbb{R}\to\mathbb{R}\), \(f_n(x)=\ln\left(x^{1/n}\right)\), converge in
\begin{examsubparts}
\item[\textnormal{(a)}]
the point-wise convergence topology?
\item[\textnormal{(b)}]
the uniform topology?
\item[\textnormal{(c)}]
compact open topology?
\end{examsubparts}
\item
State the Borsuk–Ulam Theorem for \(S^2\).
\item
Let \(X=\mathbb{R}\) and \(A=\mathbb{Z}\) be the reals and the integers respectively. Is the quotient space \(Y=X/A\) metrizable?
\item
State the Seifert–van Kampen Theorem.
\item
State the Urysohn Lemma.
\end{examproblems}
