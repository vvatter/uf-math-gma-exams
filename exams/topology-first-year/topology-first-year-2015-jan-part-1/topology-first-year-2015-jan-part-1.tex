\begin{examinstructions}
Work the following problems and show all work. Support all statements to the best of your ability.
\end{examinstructions}
\begin{examproblems}{0}{H2}
\def\ProblemHeadingText{Problem 1.}
\item
Show that there is no continuous function \(f:[0,1]\to[0,1)\) which is onto.
\def\ProblemHeadingText{Problem 2.}
\item
Show that there is no continuous function \(f:\mathbb{R}\to\{0,1\}\) which is onto.
\def\ProblemHeadingText{Problem 3.}
\item
Let~\(X\) be a metric space and let \(f:X\to X\) be a continuous function. Suppose that there is a point \(x_0\in X\) such that \(f^n(x_0)\to z\) as \(n\to\infty\). Show that \(f(z)=z\).
\def\ProblemHeadingText{Problem 4.}
\item
Show that if \(A\subset\mathbb{R}\) is connected, then~\(A\) is an interval.
\def\ProblemHeadingText{Problem 5.}
\item
Prove that if~\(X\) is compact Hausdorff, then~\(X\) is a normal space.
\def\ProblemHeadingText{Problem 6.}
\item
State and prove the Contraction Mapping Theorem.
\def\ProblemHeadingText{Problem 7.}
\item
Suppose that~\(X\) is a complete metric space that is countably infinite. Show that~\(X\) has an isolated point.
\end{examproblems}
\begin{examinstructions}
Answer the following with complete definitions or statements or short proofs.
\end{examinstructions}
\begin{examproblems}{7}{H2}
\def\ProblemHeadingText{Problem 8.}
\item
State the Intermediate Value Theorem.
\def\ProblemHeadingText{Problem 9.}
\item
State the Urysohn Lemma.
\def\ProblemHeadingText{Problem 10.}
\item
State the Tietze Extension Theorem.
\def\ProblemHeadingText{Problem 11.}
\item
State the Baire Category Theorem.
\def\ProblemHeadingText{Problem 12.}
\item
Give an example of a connected space~\(X\) that is not path connected.
\def\ProblemHeadingText{Problem 13.}
\item
Let \(n\ge 2\). Suppose that~\(A\) is a countable subset of \(\mathbb{R}^n\). Let \(X=\mathbb{R}^n\setminus A\). Is~\(X\) arcwise connected?
\def\ProblemHeadingText{Problem 14.}
\item
Suppose that \(r>0\) and that \(B_r(x)\) is a ball of radius~\(r\) centered at~\(x\) in \(\mathbb{R}^n\). Show that \(B_r(x)\) is connected.
\def\ProblemHeadingText{Problem 15.}
\item
Show that \([0,1]\) is uncountable.
\end{examproblems}
