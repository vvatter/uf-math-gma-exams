\begin{examinstructions}
Work the following problems and show all work. Support all statements to the best of your ability.
\end{examinstructions}
\begin{examproblems}{0}{H2}
\def\ProblemHeadingText{Problem 1.}
\item
Let~\(X\) be a locally connected space. Show that each component of~\(X\) is closed.
\def\ProblemHeadingText{Problem 2.}
\item
Is \(\mathbb{R}_\ell\) a Baire space? Here \(\mathbb{R}_\ell\) denotes the reals with the lower limit topology, i.e. the topology defined by the basis \(\{[a,b) \mid a,b \in \mathbb{R}\}\).
\def\ProblemHeadingText{Problem 3.}
\item
Let \(X=\mathbb{Z}\times[0,1]\subset\mathbb{R}^2\). Let \(Y=\mathbb{Z}\times(0,1]\) and \(A=\mathbb{Z}\times\{0\}\) be subspaces of~\(X\). Is the one point compactification of~\(Y\) homeomorphic to the quotient space \(X/A\)?
\def\ProblemHeadingText{Problem 4.}
\item
Show that the ordered square \(I_O^2\) is not path connected. Here \(I_O\) is the square \([0,1]\times[0,1]\) given the lexicographic order topology.
\def\ProblemHeadingText{Problem 5.}
\item
Show that the power set \(\mathcal{P}(X)\) has strictly greater cardinality than~\(X\).
\end{examproblems}
\begin{examinstructions}
Answer the following with complete definitions or statements or short proofs.
\end{examinstructions}
\begin{examproblems}{5}{H2}
\def\ProblemHeadingText{Problem 6.}
\item
Show that the circle \(S^1\) is not homeomorphic to the 2-sphere \(S^2\).
\def\ProblemHeadingText{Problem 7.}
\item
Show that the 2-sphere \(S^2\) is not homeomorphic to the plane \(\mathbb{R}^2\).
\def\ProblemHeadingText{Problem 8.}
\item
Is \(\mathbb{R}_\ell\) regular?
\def\ProblemHeadingText{Problem 9.}
\item
State the Baire Category Theorem.
\def\ProblemHeadingText{Problem 10.}
\item
Show that a connected metric space cannot be infinite countable.
\def\ProblemHeadingText{Problem 11.}
\item
State the Intermediate Value Theorem.
\def\ProblemHeadingText{Problem 12.}
\item
Give definition of a quotient map.
\def\ProblemHeadingText{Problem 13.}
\item
Are the spaces \(\mathbb{R}\times[0,\infty)\) and \([0,\infty)\times[0,\infty)\) homeomorphic?
\def\ProblemHeadingText{Problem 14.}
\item
Is every compact Hausdorff space completely regular?
\def\ProblemHeadingText{Problem 15.}
\item
Is every continuous injective map \(f:\mathbb{R}\to\mathbb{R}^2\) an embedding?
\end{examproblems}
