\begin{examinstructions}
Do 7 and only seven of the following.
\end{examinstructions}
\begin{examproblems}{0}
\item
Let \(X \subset \mathbb{R}\) consist of all numbers of the form 
\[
1+\frac{1}{\sqrt{p}}+\cdots+\frac{1}{\sqrt{p^n}}
\]
 with~\(p\) prime and \(n \in \mathbb{Z}_+\). Show that~\(X\) is countable.
\item
Assume \(f:X\to Y\) is a continuous bijection with~\(X\) compact and~\(Y\) Hausdorff. Show that~\(f\) is a homeomorphism.
\item
Prove that a compact Hausdorff space is normal.
\item
Assume that \(f:X\to Y\) is continuous and surjective.
\begin{examsubparts}
\item[\textnormal{(a)}]
If~\(X\) is Lindelöf, show that~\(Y\) is also.
\item[\textnormal{(b)}]
If~\(X\) is separable, show that~\(Y\) is also.
\end{examsubparts}
\item
Each \(A_\lambda \subset X\) is connected for \(\lambda \in \Lambda\) and \(\bigcap_{\lambda\in\Lambda} A_\lambda \ne \varnothing\), then prove that \(\bigcup_{\lambda\in\Lambda} A_\lambda\) is connected.
\item
Show that a subspace of a complete metric space is itself complete if and only if it is a closed subspace.
\item
This problem has two parts.
\begin{examsubparts}
\item[\textnormal{(a)}]
Define the metric~\(\rho\) on \([-1,1]^{\mathbb{Z}_+}\) that gives it the uniform topology.
\item[\textnormal{(b)}]
Show that in this metric the unit ball, \(B_1(\underline{0})\), is not limit point compact where \(\underline{0}\) is the sequence \(0,0,0,\ldots\).
\end{examsubparts}
\item
Prove or disprove: The real line \(\mathbb{R}_\ell\) with the lower limit topology is connected.
\end{examproblems}
