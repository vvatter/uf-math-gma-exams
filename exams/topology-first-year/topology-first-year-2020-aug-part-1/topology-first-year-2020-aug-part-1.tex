\begin{examinstructions}
Work the following problems and show all work. Support all statements to the best of your ability.
\end{examinstructions}
\begin{examproblems}{0}{H2}
\def\ProblemHeadingText{Problem 1.}
\item
Is it true that every continuous map \(f:[0,1]\to(0,1)\) has a fixed point. Justify your answer.
\def\ProblemHeadingText{Problem 2.}
\item
Show that every connected normal space having more than one point is uncountable.
\def\ProblemHeadingText{Problem 3.}
\item
Show that the set of algebraic numbers is countable (A number \(a\in\mathbb R\) is called algebraic if~\(a\) is a root of a polynomial with rational coefficients).
\def\ProblemHeadingText{Problem 4.}
\item
Show that Greek letter theta~\(\Theta\) is not homeomorphic to the Mercedes symbol~\(S\). (Formally, \(\Theta=S^1\cup[-1,1]\times\{0\}\) and \(S=S^1\cup I_1\cup I_2\cup I_3\) are subspaces of \(\mathbb R^2\) where \(S^1\) is the unit circle and \(I_k\), \(k=1,2,3\), are the closed intervals in \(\mathbb R^2\) joining the origin with the points \((0,1)\), \(\left(\frac{\sqrt3}{2},-\frac12\right)\), and \(\left(-\frac{\sqrt3}{2},-\frac12\right)\).)
\def\ProblemHeadingText{Problem 5.}
\item
Let~\(A\) be a proper subset of~\(X\) and let~\(B\) be a proper subset of~\(Y\). If~\(X\) and~\(Y\) are connected, show that \((X\times Y)-(A\times B)\) is connected.
\end{examproblems}
\begin{examinstructions}
Answer the following with complete definitions or statements or proofs.
\end{examinstructions}
\begin{examproblems}{5}{H2}
\def\ProblemHeadingText{Problem 6.}
\item
State the Extreme Value Theorem.
\def\ProblemHeadingText{Problem 7.}
\item
State the Tietze Extension Theorem.
\def\ProblemHeadingText{Problem 8.}
\item
Is the space \(I^{\mathbb N}\) separable? What about \(I^I\)? Here \(I=[0,1]\) and \(\mathbb N\) is the set of natural numbers.
\def\ProblemHeadingText{Problem 9.}
\item
Prove that the space \(X=\{(x,y)\mid xy=1\}\subset\mathbb R^2\) is locally compact? What is its one-point compactification?
\def\ProblemHeadingText{Problem 10.}
\item
Is every path connected space locally connected?
\end{examproblems}
