\begin{examinstructions}
Work the following problems and show all work. Support all statements to the best of your ability.
\end{examinstructions}
\begin{examproblems}{0}
\item
Let \(f:X\to X\) be a map of a compact metric space to itself that satisfies the following condition: \(d(f(x),f(y))<d(x,y)\).
\begin{examsubparts}
\item[\textnormal{(a)}]
Prove that~\(f\) is continuous.
\item[\textnormal{(b)}]
Show that~\(f\) has a fixed point and the fixed point is unique.
\end{examsubparts}
\item
Let \(X\subset \mathbb{R}^2\) be a countable subset. Prove that \(\mathbb{R}^2\setminus X\) is connected.
\item
Let~\(X\) be a connected normal space having more than one point. Can~\(X\) be countable?
\item
Let~\(A\) be a proper subset of~\(X\) and let~\(B\) be a proper subset of~\(Y\). If~\(X\) and~\(Y\) are connected, show that \((X\times Y)-(A\times B)\) is connected.
\item
Let \(I_O\) denote the square \([0,1]\times[0,1]\) given the lexicographic order topology. Is \(I_O\)
\begin{examsubparts}
\item[\textnormal{(a)}]
compact?
\item[\textnormal{(b)}]
connected?
\item[\textnormal{(c)}]
path connected?
\end{examsubparts}
\end{examproblems}
\begin{examinstructions}
Answer the following with complete definitions or statements or short proofs.
\end{examinstructions}
\begin{examproblems}{5}
\item
Give definition of Lebesgue number.
\item
Show that the power set \(P(X)\) has strictly greater cardinality than~\(X\).
\item
Show that the 2-sphere is not homeomorphic to
\begin{examsubparts}
\item[\textnormal{(a)}]
the circle \(S^1\);
\item[\textnormal{(b)}]
the 2-plane \(\mathbb{R}^2\).
\end{examsubparts}
\item
Show that every compact Hausdorff space is regular.
\item
Give definition of a quotient map. Let 
\[
Z=\{(x,y)\in\mathbb{R}^2\mid xy=1\}\cup(\{0\}\times\mathbb{R})\subset\mathbb{R}\times\mathbb{R}
\]
 be given the subspace topology. Is the projection \(f:Z\to\mathbb{R}\) to the first coordinate, \(f(x,y)=x\), a quotient map?
\end{examproblems}
