\begin{examinstructions}
Work the following problems and show all work. Support all statements to the best of your ability.
\end{examinstructions}
\begin{examproblems}{0}
\item
Prove that the circle \(S^1\) is not homeomorphic to the closed 2-disk \(B^2\).
\item
Let \(A\subset \mathbb{R}^2\) be an infinite countable subspace.
\begin{examsubparts}
\item[\textnormal{(a)}]
Can~\(A\) be connected?
\item[\textnormal{(b)}]
Is \(\mathbb{R}^2\setminus A\) connected?
\end{examsubparts}
\item
Let \(f:X\to\mathbb{R}^2\) be continuous injective map. Is~\(f\) an embedding if
\begin{examsubparts}
\item[\textnormal{(a)}]
\(X=(0,1)\)?
\item[\textnormal{(b)}]
\(X=[0,1]\)?
\item[\textnormal{(c)}]
\(X=[0,1)\)?
\end{examsubparts}
\item
Show that a continuous injective map \(f:(0,1)\to\mathbb{R}\) is an embedding.
\item
Show that every compact Hausdorff space is normal.
\end{examproblems}
\begin{examinstructions}
Answer the following with complete definitions or statements or short proofs.
\end{examinstructions}
\begin{examproblems}{5}
\item
Describe all connected subsets of the real line~\(\mathbb{R}\).
\item
State the Contraction Mapping Theorem.
\item
Let \(\mathbb{R}_\ell\) denote the reals with the lower limit topology, i.e. topology defined by the basis \(\{[a,b)\mid a,b\in\mathbb{R}\}\). Is \(\mathbb{R}_\ell\)
\begin{examsubparts}
\item[\textnormal{(a)}]
connected?
\item[\textnormal{(b)}]
regular?
\end{examsubparts}
\item
State the Baire Category Theorem.
\item
Show that a connected metric space cannot be infinite countable.
\item
State the Urysohn Lemma.
\item
Give definition of a quotient map.
\item
Are the spaces \((0,1)\times[0,1)\) and \([0,1)\times[0,1)\) homeomorphic?
\item
Describe the one-point compactification of \(X=(0,1)\cup(2,3)\).
\item
Let~\(X\) be a locally connected space. Is every component of~\(X\) closed?
\end{examproblems}
