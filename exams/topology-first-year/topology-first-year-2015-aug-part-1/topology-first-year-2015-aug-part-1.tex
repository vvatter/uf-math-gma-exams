\begin{examinstructions}
Answer all questions and work all problems.
\end{examinstructions}
\begin{examproblems}{0}
\item
Suppose that \(f:X\to Y\) is continuous and onto. Show that if~\(X\) is compact, then~\(Y\) is also compact.
\item
Suppose that \(A\subset X\) is connected. Show that the closure of~\(A\), \(\overline{A}\), is also connected.
\item
State and prove the Banach Fixed-Point Theorem.
\item
Suppose that~\(X\) is a compact metric space. Show that if \(\{x_i\}\) is a sequence of points in~\(X\), then there is a point \(z\in X\) and a subsequence, \(\{x_{i_j}\}\), such that \(x_{i_j}\to z\).
\item
Give an example of a sequence of functions \(f_i:[0,1]\to[0,1]\) such that \(f_i(x)\to g(x)\) for all \(x\in[0,1]\), but such that \(g:[0,1]\to[0,1]\) is not continuous.
\item
Suppose that~\(X\) is a set and that \(f:X\to 2^X\). Show that~\(f\) cannot be onto.
\item
Suppose that~\(X\) is a complete metric space that is countably infinite. Show that~\(X\) has an isolated point, i.e., there is a point \(x\in X\) such that \(\{x\}\) is open in~\(X\).
\end{examproblems}
\begin{examinstructions}
Answer the following with complete definitions or statements or short proofs.
\end{examinstructions}
\begin{examproblems}{7}
\item
State The Urysohn Lemma.
\item
Show that \([0,1)\) is not compact.
\item
State the Tietze Extension Theorem.
\item
State the Tychonoff Theorem.
\item
State the Alexander Subbase Theorem.
\item
Give an example of a Hausdorff space that is not normal.
\end{examproblems}
