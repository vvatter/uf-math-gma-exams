\begin{examinstructions}
Do seven (7) and only seven of the following.
\end{examinstructions}
\begin{examproblems}{0}{H2}
\def\ProblemHeadingText{Problem 1.}
\item
Prove that a compact Hausdorff space is normal.
\def\ProblemHeadingText{Problem 2.}
\item
Say \(A \subset X\) is dense if \(\overline{A}=X\).
\begin{examsubparts}
\item[\textnormal{(a)}]
Show that~\(A\) is dense in~\(X\) if and only if every nonempty open subset~\(V\) in~\(X\) satisfies \(V\cap A\ne\emptyset\).
\item[\textnormal{(b)}]
Assume that~\(X\) and~\(Y\) are topological spaces with~\(Y\) Hausdorff and~\(A\) is dense in~\(X\). Suppose that \(f:X\to Y\) and \(g:X\to Y\) are continuous functions with \(f(a)=g(a)\) for all \(a\in A\). Prove that \(f(x)=g(x)\) for all \(x\in X\).
\end{examsubparts}
\def\ProblemHeadingText{Problem 3.}
\item
Let \(B\subset\mathbb{R}^{\mathbb{Z}_+}\) be defined as
\[
B=\prod_{i\in\mathbb{Z}_+}(-1,1)=(-1,1)\times(-1,1)\times(-1,1)\times\cdots.
\]
\begin{examsubparts}
\item[\textnormal{(a)}]
Is~\(B\) open in the box topology?
\item[\textnormal{(b)}]
Is~\(B\) open in the uniform topology?
\item[\textnormal{(c)}]
Is~\(B\) open in the product topology?
\end{examsubparts}
\def\ProblemHeadingText{Problem 4.}
\item
If \((X,d)\) is a metric space and \(A\subset X\), for \(x\in X\) define
\[
d(x,A)=\inf\{d(x,a):a\in A\}.
\]
\begin{examsubparts}
\item[\textnormal{(a)}]
Show that the function \(f:X\to[0,\infty)\) defined by \(f(x)=d(x,A)\) is continuous.
\item[\textnormal{(b)}]
Show that \(x\in\overline{A}\) if and only if \(f(x)=0\).
\item[\textnormal{(c)}]
If~\(A\) is compact then for any \(x\in X\) there exists an \(a\in A\) with \(d(x,A)=d(x,a)\).
\end{examsubparts}
\def\ProblemHeadingText{Problem 5.}
\item
Show that \((X,d)\) is complete if and only if any nested family of nonempty closed sets \(A_n\) with \(\operatorname{diam}(A_n)\to0\) has \(\bigcap A_n\ne\emptyset\).
\def\ProblemHeadingText{Problem 6.}
\item
Assume that \(f:X\to Y\) is continuous and surjective.
\begin{examsubparts}
\item[\textnormal{(a)}]
If~\(X\) is Lindelöf, show that~\(Y\) is also.
\item[\textnormal{(b)}]
If~\(X\) is separable, show that~\(Y\) is also.
\end{examsubparts}
\def\ProblemHeadingText{Problem 7.}
\item
If~\(X\) and~\(Y\) are connected topological spaces, show that \(X\times Y\) is connected.
\def\ProblemHeadingText{Problem 8.}
\item
In the real line~\(\mathbb{R}\) let~\(\mathcal{T}\) be the collection of all subsets~\(A\) of~\(\mathbb{R}\) such that \(\mathbb{R}-A\) is either finite or all of~\(\mathbb{R}\).
\begin{examsubparts}
\item[\textnormal{(a)}]
Prove that \((\mathbb{R},\mathcal{T})\) is topological space.
\item[\textnormal{(b)}]
Is this topology Hausdorff?
\item[\textnormal{(c)}]
To which point or points does the sequence \(x_n=1/n^2\) converge?
\end{examsubparts}
\end{examproblems}
