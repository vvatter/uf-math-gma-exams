\begin{examinstructions}
For the first five problems, show all your work and support all statements.
\end{examinstructions}
\begin{examproblems}{0}
\item
Show there is no surjective function \(f:X\to\mathcal{P}(X)\), where \(\mathcal{P}(X)\) is the power set of~\(X\).
\item
Show that if~\(X\) is a compact Hausdorff space, then~\(X\) is a normal space.
\item
Let \((X,d)\) be a complete metric space. Let \(f:X\to X\) and suppose there is some \(c<1\) with \(d(f(x),f(y))\leq c\cdot d(x,y)\) for all \(x,y\in X\). Prove there exists a unique \(x\in X\) with \(f(x)=x\).
\item
Let~\(A\) be a proper subset of~\(X\), and let~\(B\) be a proper subset of~\(Y\). If~\(X\) and~\(Y\) are connected, show that \((X\times Y)-(A\times B)\) is connected.
\item
Let \((X,d)\) be a metric space. Suppose there exists some \(\varepsilon>0\) such that for all \(x\in X\), the closure \(\overline{B_\varepsilon(x)}\) of the~\(\varepsilon\)-ball about~\(x\) is compact. Prove that~\(X\) is complete.
\end{examproblems}
\begin{examinstructions}
Answer the following problems with complete definitions, complete statements, an example, or a short proof.
\end{examinstructions}
\begin{examproblems}{5}
\item
Let \(\{A_n\}_{n\in\mathbb{Z}_+}\) be a countable collection of countable sets. Prove that the union \(\bigcup_{n\in\mathbb{Z}_+}A_n\) is countable.
\item
Give an example of a topological space that is path-connected but not locally path-connected.
\item
Let \(\{X_\alpha\}_{\alpha\in A}\) be a collection of topological spaces. What is the product topology on \(\prod_{\alpha\in A}X_\alpha\)?
\item
Show that \(f:\mathbb{R}\to\mathbb{R}^\omega\) defined by \(f(x)=(x,x,x,\ldots)\) is not continuous if \(\mathbb{R}^\omega\) has the box topology.
\item
If \(f:X\to Y\) is a continuous bijection with~\(X\) compact and~\(Y\) Hausdorff, then show that \(f^{-1}\) is continuous.
\item
Give the definition of a quotient map.
\item
Show that \([0,1]\) and \((0,1)\) are not homeomorphic.
\item
Using the fact that there exists a continuous surjective function \([0,1]\to[0,1]^2\), show there is a continuous surjective function \([0,1]\to[0,1]^3\).
\item
State the Baire Category Theorem.
\item
Let~\(X\) be a locally compact Hausdorff space. Define the topology on \(Y=X\cup\{\infty\}\) that makes~\(Y\) the one-point compactification of~\(X\).
\end{examproblems}
