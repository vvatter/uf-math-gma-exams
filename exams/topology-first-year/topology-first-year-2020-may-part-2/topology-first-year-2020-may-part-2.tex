\begin{examinstructions}
Work the following problems and show all work. Support all statements to the best of your ability.
\end{examinstructions}
\begin{examproblems}{0}
\item
Show that \(S^2\) is not homeomorphic to \(S^3\).
\item
Prove that the set of irrational numbers \(\mathcal I\subset\mathbb R\) cannot be presented as a countable union of closed sets each of which has empty interior in \(\mathcal I\).
\item
Show that for every continuous map \(f:S^1\to S^1\) there is \(n\in\mathbb Z\) such that~\(f\) is homotopic to \(z^n:S^1\to S^1\).
\item
Show that every map \(f:S^2\to T\) of the 2-sphere to the torus~\(T\) is null-homotopic.
\item
Show that the torus~\(T\) is not homeomorphic to the surface of genus 2, \(M_2=T\mathbin{\#}T\).
\end{examproblems}
\begin{examinstructions}
Answer the following with complete definitions or statements or proofs.
\end{examinstructions}
\begin{examproblems}{5}
\item
Does the sequence \(f_n:\mathbb R\to\mathbb R\), \(f_n(x)=\sqrt{x}/n\), converge in
\begin{examsubparts}
\item[\textnormal{(a)}]
the point-wise convergence topology?
\item[\textnormal{(b)}]
the uniform topology?
\item[\textnormal{(c)}]
compact open topology?
\end{examsubparts}
\item
State the Borsuk–Ulam Theorem for \(S^2\).
\item
Is the space \(I^I\) separable where \(I=[0,1]\)?
\item
State the Seifert–van Kampen Theorem.
\item
State and prove the Tychonoff Theorem.
\end{examproblems}
