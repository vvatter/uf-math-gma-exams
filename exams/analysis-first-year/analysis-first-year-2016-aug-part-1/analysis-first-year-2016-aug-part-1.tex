\begin{examinstructions}
Answer exactly FOUR questions. Write solutions in a neat and logical fashion, giving complete reasons for all steps.
\end{examinstructions}
\begin{examproblems}{0}
\item
Let \((a_n)_{n=1}^{\infty}\) and \((b_n)_{n=1}^{\infty}\) be bounded real sequences and let \((c_n)_{n=1}^{\infty}\) be the sequence 
\[
a_1,b_1,a_2,b_2,\ldots
\]
 Prove that 
\[
\limsup_{n\to\infty}c_n\leq \max\left(\limsup_{n\to\infty}a_n,\limsup_{n\to\infty}b_n\right)
\]
 and that if \((a_n)_{n=1}^{\infty}\) and \((b_n)_{n=1}^{\infty}\) converge to~\(\gamma\), then so does \((c_n)_{n=1}^{\infty}\).
\item
State the definition of a perfect set and prove that the standard Cantor set in \([0,1]\) is a perfect set.
\item
Let \(f:X\to Y\) be a map between metric spaces and let 
\[
G_f=\{(x,f(x)):x\in X\}\subseteq X\times Y
\]
 be its graph. Prove that if \(G_f\) is compact then~\(f\) is continuous.
\item
Let~\(f\) be a uniformly continuous real-valued function on the bounded open interval \((a,b)\). Prove that the real limit \(\lim_{t\to b^-}f(t)\) exists. Show by example that we may not replace uniformly by bounded without jeopardizing this conclusion.
\item
Let the real-valued function~\(f\) be continuous on \([0,2]\) and twice-differentiable on \((0,2)\). Assume that \(f(0)=0\), that \(f(1)=1\), and that \(f(2)=2\). Prove that there exists \(u\in(0,2)\) such that \(f''(u)=0\).
\end{examproblems}
