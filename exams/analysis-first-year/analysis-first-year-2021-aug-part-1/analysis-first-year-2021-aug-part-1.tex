\begin{examinstructions}
Answer FOUR questions in detail. State carefully any results used without proof.
\end{examinstructions}
\begin{examproblems}{0}
\item
Let \(f:(0,\infty)\to\mathbb{R}\) be continuous. Prove that if \(\limsup_{n\to\infty} f(n)=1\) and \(\liminf_{n\to\infty} f(n)=-1\) then~\(f\) has infinitely many zeros.
\item
Let \(f:X\to Y\) be a continuous bijection between metric spaces. For each of the following statements, give a proof or a counterexample, as appropriate:
\begin{examsubparts}
\item[\textnormal{(a)}]
If~\(X\) is compact then \(f^{-1}\) is continuous;
\item[\textnormal{(b)}]
if~\(Y\) is compact then \(f^{-1}\) is continuous.
\end{examsubparts}
\item
The function \(f:\mathbb{R}\to\mathbb{R}\) satisfies \(\lim_{x\to-\infty}f(x)=0\) and \(\lim_{x\to\infty}f(x)=0\). Prove that if~\(f\) is continuous then it is uniformly continuous.
\item
Let \((a_n)_{n=0}^{\infty}\) be a sequence of positive reals. Prove that if the series \(\sum_{n=0}^{\infty}a_n\) converges then so does the series \(\sum_{n=0}^{\infty}\sqrt{a_na_{n+1}}\). Prove also that the converse holds when \((a_n)_{n=0}^{\infty}\) is decreasing.
\item
Let the function~\(f\) be defined on an open interval about~\(0\); assume that \(f(0)=0\) and that~\(f\) is differentiable at~\(0\). When~\(n\) is a positive integer, determine the value of the limit 
\[
\lim_{x\to 0}\frac{1}{x}\left[f(x)+f\!\left(\frac{x}{2}\right)+f\!\left(\frac{x}{3}\right)+\cdots+f\!\left(\frac{x}{n}\right)\right].
\]
\end{examproblems}
