\begin{examinstructions}
Answer FOUR questions in detail. State carefully any results used without proof.
\end{examinstructions}
\begin{examproblems}{0}
\item
Let \((a_n)\) and \((b_n)\) be bounded real sequences. Prove that 
\[
\limsup(a_n+b_n)\leq \limsup a_n+\limsup b_n
\]
 with equality when \((a_n)\) converges.
\item
Let~\(A\) be a subset of the metric space \((M,d)\) and define \(d_A:M\to\mathbb{R}\) by \(d_A(x)=\inf\{d(a,x):a\in A\}\). Prove that:
\begin{examsubparts}
\item[\textnormal{(i)}]
\(d_A\) is uniformly continuous;
\item[\textnormal{(ii)}]
\(\{x\in M:d_A(x)=0\}=\overline{A}\).
\end{examsubparts}
\item
Let~\(X\) and~\(Y\) be metric spaces; let \(f:X\to Y\) be a continuous bijection, with inverse \(g:Y\to X\). For each of the following statements, provide a proof if true or a counterexample (with justification) if false:
\begin{examsubparts}
\item[\textnormal{(i)}]
if~\(X\) is compact then~\(g\) is continuous;
\item[\textnormal{(ii)}]
if~\(Y\) is compact then~\(g\) is continuous.
\end{examsubparts}
\item
Let~\(X\) and~\(Y\) be metric spaces; let \(f:X\to Y\) be a continuous bijection, with uniformly continuous inverse \(g:Y\to X\). For each of the following statements, provide a proof if true or a counterexample if false:

\par
[Be sure to demonstrate uniform continuity of~\(g\) in any counterexample.]
\begin{examsubparts}
\item[\textnormal{(i)}]
if~\(X\) is complete then~\(Y\) is complete;
\item[\textnormal{(ii)}]
if~\(Y\) is complete then~\(X\) is complete.
\end{examsubparts}
\item
The function \(f:\mathbb{R}\to\mathbb{R}\) is differentiable, with \(|f'|\leq 1\) everywhere. Prove that if \(\varepsilon>0\) is sufficiently small (how small?) then the function \(F:\mathbb{R}\to\mathbb{R}\) defined by \(F(x)=x+\varepsilon f(x)\) is injective.
\end{examproblems}
