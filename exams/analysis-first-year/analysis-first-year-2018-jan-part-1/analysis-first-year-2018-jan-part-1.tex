\begin{examinstructions}
Answer FOUR questions in detail. State carefully any results used without proof.
\end{examinstructions}
\begin{examproblems}{0}
\item
Let \((a_n:n\ge 0)\) be a bounded real sequence. For each \(N\ge 0\) define \(\bar a_N=\sup\{a_n:n\ge N\}\) and let \(s=\inf\{\bar a_N:N\ge 0\}\). Prove:
\begin{examsubparts}
\item[\textnormal{(i)}]
if \(s<t\) then \(a_n<t\) eventually, in the sense \((\exists N)(\forall n\ge N)(a_n<t)\);
\item[\textnormal{(ii)}]
if \(s>r\) then \(a_n>r\) frequently, in the sense \((\forall N)(\exists n\ge N)(a_n>r)\).
\end{examsubparts}
\item
Let~\(X\) be a metric space. Assume that there exists \(\delta>0\) such that the distance between distinct points of~\(X\) is always~\(\delta\) or greater. Determine precisely which subsets of~\(X\) are open, which closed, which compact and which connected.
\item
Let \(f:X\to\mathbb{R}\) have the property that if~\(a\) is any real number then \(U(a)=\{x:f(x)<a\}\) is an open subset of the compact metric space~\(X\).
\begin{examsubparts}
\item[\textnormal{(i)}]
Prove that~\(f\) is bounded above on~\(X\).
\item[\textnormal{(ii)}]
Prove that \(s=\sup\{f(x):x\in X\}\) is a value of~\(f\). For (ii): suppose~\(s\) not a value and consider the function \(F=1/(s-f)\).
\end{examsubparts}
\item
Let \(f:X\to Y\) be a bijection between metric spaces; assume that~\(Y\) is complete and that \(f^{-1}\) is continuous.
\begin{examsubparts}
\item[\textnormal{(i)}]
Show that if~\(f\) is uniformly continuous then~\(X\) is complete.
\item[\textnormal{(ii)}]
Show that if~\(f\) is only continuous then~\(X\) can fail to be complete.
\end{examsubparts}
\item
Let \(f:(a,b)\to\mathbb{R}\) be differentiable and assume that its derivative \(f'\) is bounded. Prove that the right-hand limit \(\lim_{t\to a+}f(t)\) exists.
\end{examproblems}
