\begin{examinstructions}
Answer FOUR questions in detail. State carefully any results used without proof.
\end{examinstructions}
\begin{examproblems}{0}
\item
Prove
\begin{examsubparts}
\item[\textnormal{(a)}]
that every real sequence has a monotonic subsequence
\item[\textnormal{(b)}]
that every bounded real sequence has a convergent subsequence.
\end{examsubparts}
\item
Let~\(T\) be a two-point metric space. Prove that the metric space~\(X\) is connected if and only if each continuous function from~\(X\) to~\(T\) is constant.
\item
Let \(X \subseteq \mathbb{R}\) and consider the following statements: Does (a) imply (b)? Does (b) imply (a)? Proof or counterexample for each.
\begin{examsubparts}
\item[\textnormal{(a)}]
each continuous function from~\(X\) to~\(\mathbb{R}\) is bounded;
\item[\textnormal{(b)}]
\(X\) is compact.
\end{examsubparts}
\item
Let \(f : \mathbb{R} \to \mathbb{R}\) be uniformly continuous and let \(B \subseteq \mathbb{R}\) be bounded. Must \(f(B)\) be bounded? Proof or counterexample.
\item
The continuous function \(f : \mathbb{R} \to \mathbb{R}\) is differentiable except perhaps at the point~\(a\). Decide whether~\(f\) is necessarily differentiable at~\(a\) if it is further assumed that \(f'(t) \to 0\) as \(t \to a\).
\end{examproblems}
