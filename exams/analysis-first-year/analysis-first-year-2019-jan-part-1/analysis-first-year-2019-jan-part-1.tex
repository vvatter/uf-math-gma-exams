\begin{examinstructions}
Do exactly 4 problems. Work must be presented in a neat and logical fashion in order to receive credit. Do not leave any gaps. When a theorem is used in a proof it must be precisely stated.
\end{examinstructions}
\begin{examproblems}{0}{H2}
\def\ProblemHeadingText{Problem 1.}
\item
Let \((s_n)_{n=1}^{\infty}\) be a bounded sequence of real numbers. Consider the sequence \((\sigma_n)\) of averages
\[
\sigma_n=\frac{s_1+s_2+\cdots+s_n}{n}.
\]
 Prove that the sequence \((\sigma_n)\) is bounded and
\[
\limsup \sigma_n\leq \limsup s_n.
\]
\def\ProblemHeadingText{Problem 2.}
\item
Let \(K_1\supseteq K_2\supseteq K_3\supseteq\cdots\) be a decreasing family of nonempty sets in a metric space~\(X\).
\begin{examsubparts}
\item[\textnormal{a)}]
Prove that if each \(K_n\) is compact, then \(\bigcap_n K_n\) is nonempty.
\item[\textnormal{b)}]
Give an example to show that \(\bigcap_n K_n\) can be empty if the \(K_n\) are only assumed closed.
\end{examsubparts}
\def\ProblemHeadingText{Problem 3.}
\item
Let \(X,Y\) be metric spaces and \(f:X\to Y\) a function. Prove that if~\(f\) is uniformly continuous on~\(X\) and \((x_n)\) is a Cauchy sequence in~\(X\), then \((f(x_n))\) is a Cauchy sequence in~\(Y\).
\def\ProblemHeadingText{Problem 4.}
\item
Let~\(X\) be a metric space and suppose that every continuous function \(f:X\to\mathbb{Z}\) is constant. Prove that~\(X\) is connected.
\def\ProblemHeadingText{Problem 5.}
\item
Suppose~\(f\) is differentiable in \((a,b)\) and \(f'(x)>0\) for all \(x\in(a,b)\). Let~\(g\) be the inverse function to~\(f\). Prove that for each \(x_0\in(a,b)\), the function~\(g\) is differentiable at \(f(x_0)\) and
\[
g'(f(x_0))=\frac{1}{f'(x_0)}.
\]
\end{examproblems}
