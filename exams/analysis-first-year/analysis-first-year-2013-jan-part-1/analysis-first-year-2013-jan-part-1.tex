\begin{examinstructions}
Answer FOUR questions. Write in a neat and logical fashion, giving complete reasons for all steps.
\end{examinstructions}
\begin{examproblems}{0}
\item
Let \((a_n)_{n=1}^{\infty}\) be a real sequence and for \(n \geq 1\) define 
\[
\alpha_n=(a_1+\cdots+a_n)/n.
\]
 Prove that 
\[
\limsup_{n\to\infty}\alpha_n\leq\limsup_{n\to\infty}a_n
\]
 and show by example that the inequality can be strict.
\item
Let~\(X\) and~\(Y\) be metric spaces; give \(X\times Y\) the metric defined by 
\[
d((x,y),(a,b))=d_X(x,a)+d_Y(y,b).
\]
\begin{examsubparts}
\item[\textnormal{(i)}]
If~\(X\) and~\(Y\) are compact, is \(X\times Y\) compact? Proof or counterexample.
\item[\textnormal{(ii)}]
If~\(X\) and~\(Y\) are complete, is \(X\times Y\) complete? Proof or counterexample.
\end{examsubparts}
\item
Let \(f:X\to X\) be a continuous map from a metric space to itself; assume that~\(f\) has no fixed points. Prove that if~\(X\) is compact then there exists \(\varepsilon>0\) such that \(d(x,f(x))\geq\varepsilon\) for each \(x\in X\).
\item
For a map \(f:X\to Y\) between metric spaces, consider the statements: Of the implications (a)~\(\Rightarrow\) (b) and (b)~\(\Rightarrow\) (a), one is false and the other true; give a proof of the true and a counterexample for the false.
\begin{examsubparts}
\item[\textnormal{(a)}]
\(f\) is uniformly continuous;
\item[\textnormal{(b)}]
\(f\) maps Cauchy sequences to Cauchy sequences.
\end{examsubparts}
\item
Let \(f:(0,\infty)\to\mathbb{R}\) be differentiable and let~\(b\) be a real number. Prove that if \(f'(t)\to b\) as \(t\to\infty\) then \(f(t)/t\to b\) as \(t\to\infty\).
\end{examproblems}
