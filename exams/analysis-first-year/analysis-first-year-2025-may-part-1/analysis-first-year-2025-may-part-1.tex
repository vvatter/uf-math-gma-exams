\begin{examinstructions}
Answer FOUR questions in detail. State carefully any results used without proof.
\end{examinstructions}
\begin{examproblems}{0}
\item
Let~\(A\) be a subset of the metric space~\(M\). Decide (with proof or counterexample) whether the following statements are equivalent:
\begin{examsubparts}
\item[\textnormal{(i)}]
\(A\) is open;
\item[\textnormal{(ii)}]
for each \(X\subseteq M\), if \(A\cap X=\emptyset\) then \(A\cap\overline{X}=\emptyset\).
\end{examsubparts}
\item
The function \(f:X\to Y\) between metric spaces is distance-preserving. Must \(f(X)\) be closed in~\(Y\) in each of the following cases? If the answer is Yes, give a proof; if the answer is No, give a counterexample.
\begin{examsubparts}
\item[\textnormal{(i)}]
\(X\) is compact.
\item[\textnormal{(ii)}]
\(X\) is complete.
\end{examsubparts}
\item
Let \(f:X\to Y\) be a continuous bijection, \(g:Y\to X\) its inverse. Must~\(g\) be continuous in each of the following cases? Proof or counterexample, as appropriate.
\begin{examsubparts}
\item[\textnormal{(i)}]
\(X\) is compact.
\item[\textnormal{(ii)}]
\(Y\) is compact.
\end{examsubparts}
\item
The function \(f:\mathbb{R}\to\mathbb{R}\) is continuous and satisfies 
\[
\liminf f(n)<0<\limsup f(n)
\]
 where~\(n\) runs through the natural numbers. Prove that each of \(f^{-1}(-\infty,0)\) and \(f^{-1}(0,\infty)\) is infinite.
\item
Let \(f:\mathbb{R}\to\mathbb{R}\) be such that 
\[
|f(x)-f(y)|\leqslant |x-y|^a
\]
 for some \(a>1\) and all \(x,y\in\mathbb{R}\). Deduce as much as possible about~\(f\).
\end{examproblems}
