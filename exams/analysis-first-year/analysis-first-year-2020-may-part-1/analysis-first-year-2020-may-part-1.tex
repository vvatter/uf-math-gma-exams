\begin{examinstructions}
Answer FOUR questions in detail. State carefully any results used without proof.
\end{examinstructions}
\begin{examproblems}{0}
\item
Let \((a_n)_{n=0}^{\infty}\) be a bounded real sequence. Write~\(L\) for the set of all real numbers~\(\ell\) such that \((\exists N)(\forall n>N)\;\ell<a_n\) and write~\(U\) for the set of all real numbers~\(u\) such that \((\exists N)(\forall n>N)\;a_n<u\).
\begin{examsubparts}
\item[\textnormal{(i)}]
Prove that \(\sup L\leq\inf U\).
\item[\textnormal{(ii)}]
Give an example of a convergent sequence for which the inequality in (i) is strict, or prove that no such sequence exists.
\end{examsubparts}
\item
Let~\(U\) be a subset of the metric space~\(X\). Does either of the following conditions imply the other? Prove or disprove, as appropriate.
\begin{examsubparts}
\item[\textnormal{(i)}]
\(U\) is open.
\item[\textnormal{(ii)}]
For any \(A\subseteq X\), if~\(U\) meets the closure of~\(A\) then~\(U\) meets~\(A\) itself.
\end{examsubparts}
\item
Let~\(X\) and~\(Y\) be metric spaces. Decide whether the following statements are true or false.
\begin{examsubparts}
\item[\textnormal{(i)}]
If \((x_n)_{n=0}^{\infty}\) converges to~\(x\) in~\(X\) then \(\{x\}\cup\{x_n:n\geq 0\}\) is compact.
\item[\textnormal{(ii)}]
The function \(f:X\to Y\) is continuous if its restriction \(f|_K\) to each compact subset \(K\subseteq X\) is continuous.
\end{examsubparts}
\item
Let \(f:X\to Y\) be a continuous bijection with inverse \(g:Y\to X\) and assume that~\(X\) is complete.
\begin{examsubparts}
\item[\textnormal{(i)}]
Show by example that if~\(g\) is continuous then~\(Y\) need not be complete.
\item[\textnormal{(ii)}]
Prove that if~\(g\) is uniformly continuous then~\(Y\) must be complete.
\end{examsubparts}
\item
Let \(f:(0,\infty)\to\mathbb{R}\) be differentiable and assume \(\lim_{x\to\infty}f'(x)=L\in\mathbb{R}\).
\begin{examsubparts}
\item[\textnormal{(i)}]
When~\(a\) is a positive real number, calculate 
\[
\lim_{x\to\infty}\frac{f(x+a)-f(x)}{a}.
\]
\item[\textnormal{(ii)}]
Assume further that \(f(x)\to A\in\mathbb{R}\) as \(x\to\infty\) and calculate~\(L\).
\end{examsubparts}
\end{examproblems}
