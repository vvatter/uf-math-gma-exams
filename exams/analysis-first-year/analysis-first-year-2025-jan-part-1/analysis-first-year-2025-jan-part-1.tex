\begin{examinstructions}
Answer FOUR questions in detail. State carefully any results used without proof.
\end{examinstructions}
\begin{examproblems}{0}
\item
Let \((a_n)\) be a bounded sequence of strictly positive real numbers. Prove that 
\[
\limsup (a_n^{1/n}) \leq \limsup (a_{n+1}/a_n).
\]
 Hence prove that if \(a_{n+1}/a_n \to \ell \in \mathbb{R}\) then \(a_n^{1/n} \to \ell\) (as \(n \to \infty\)).
\item
The diameter~\(D\) of the bounded metric space~\(X\) is defined by 
\[
D=\sup\{d(a,b):a,b\in X\}.
\]
 Prove that if~\(X\) is compact then \(D=d(a_0,b_0)\) for some \(a_0,b_0\in X\).
\item
The compact metric space~\(X\) contains \(A_1\supseteq A_2\supseteq\cdots\) as a decreasing sequence of closed sets.
\begin{examsubparts}
\item[\textnormal{(i)}]
Prove that if each \(A_n\) is nonempty then so is the intersection \(A=\bigcap_{n=1}^{\infty}A_n\).
\item[\textnormal{(ii)}]
Prove that if \(U\subseteq X\) is open and contains~\(A\) then~\(U\) contains some \(A_n\).
\end{examsubparts}
\item
Let \(f:X\to Y\) be a map between metric spaces and consider the statements. Does (a) imply (b)? Does (b) imply (a)? In each case, give a proof if so, a counterexample if no.
\begin{examsubparts}
\item[\textnormal{(a)}]
if \((x_n)\) is any Cauchy sequence in~\(X\) then \((f(x_n))\) is Cauchy in~\(Y\);
\item[\textnormal{(b)}]
\(f\) is uniformly continuous.
\end{examsubparts}
\item
Let \(\ell\in\mathbb{R}\), let \(f:\mathbb{R}\to\mathbb{R}\), and assume that \(f'(t)\to 0\) as \(\mathbb{R}\ni t\to\infty\). Prove that if \(f(n)\to\ell\) as \(\mathbb{N}\ni n\to\infty\) then \(f(t)\to\ell\) as \(\mathbb{R}\ni t\to\infty\).
\end{examproblems}
