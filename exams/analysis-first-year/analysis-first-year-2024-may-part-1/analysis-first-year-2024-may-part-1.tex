\begin{examinstructions}
Answer FOUR questions in detail. State carefully any results used without proof.
\end{examinstructions}
\begin{examproblems}{0}
\item
Assume without proof that if \((a_n)\) and \((b_n)\) are bounded real sequences then 
\[
\limsup(a_n+b_n)\leq \limsup a_n+\limsup b_n.
\]
\begin{examsubparts}
\item[\textnormal{(i)}]
Show by example that this inequality can be strict.
\item[\textnormal{(ii)}]
Prove that equality holds if \((a_n)\) is convergent.
\end{examsubparts}
\item
Let~\(A\) be a subset of the metric space~\(M\). Assume that~\(A\) satisfies the following condition: 
\[
\forall X\subseteq M\ \bigl(A\cap X=\varnothing\Rightarrow A\cap\overline{X}=\varnothing\bigr).
\]
 Does it follow that~\(A\) is open? Proof or counterexample.
\item
Let~\(A\) and~\(B\) be nonempty compact subsets of the metric space~\(M\). Prove that there exist \(a\in A\) and \(b\in B\) such that 
\[
d(a,b)=\inf\{d(x,y):x\in A,y\in B\}.
\]
\item
Does there exist a one-to-one continuous function from the unit square \(\square=[0,1]\times[0,1]\) to~\(\mathbb{R}\)? If so, exhibit one; if no, prove not.
\item
Let \(f:(0,\infty)\to\mathbb{R}\) be differentiable and let \(b\in\mathbb{R}\). Prove that if \(f'(t)\to b\) as \(t\to\infty\) then \(f(t)/t\to b\) as \(t\to\infty\). [It might help to choose and fix \(a>0\).]
\end{examproblems}
