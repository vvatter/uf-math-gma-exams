\begin{examinstructions}
Answer seven problems. You should indicate which problems you wish to have graded. Write your answers clearly in complete English sentences. You may quote results (within reason) as long as you state them clearly.
\end{examinstructions}
\begin{examproblems}{0}
\item
This problem has three parts.
\begin{examsubparts}
\item[\textnormal{(a)}]
Give the definition of a solvable group.
\item[\textnormal{(b)}]
Prove that the homomorphic image of a solvable group is solvable.
\item[\textnormal{(c)}]
Let~\(F\) be a free group on a set~\(S\) with \(|S|>1\). Prove that~\(F\) is not solvable.
\end{examsubparts}
\item
This problem has three parts.
\begin{examsubparts}
\item[\textnormal{(a)}]
Give the definition of an epimorphism in a category~\(C\).
\item[\textnormal{(b)}]
Prove that the composition of two epimorphisms is an epimorphism.
\item[\textnormal{(c)}]
Let~\(C\) be the category whose objects are commutative rings with 1 and whose morphisms are unitary ring homomorphisms. Prove that the natural map \(\mathbb{Z}\to\mathbb{Q}\) is an epimorphism in~\(C\).
\end{examsubparts}
\item
Let \(L/K\) be an algebraic field extension. Let~\(S\) denote the set of elements of~\(L\) which are separable over~\(K\).
\begin{examsubparts}
\item[\textnormal{(a)}]
Prove that~\(S\) is a field.
\item[\textnormal{(b)}]
Prove that for every \(\alpha\in L\) the extension \(S(\alpha)/S\) is purely inseparable.
\end{examsubparts}
\item
Give an example of a polynomial \(f(X)\in\mathbb{Q}[X]\) whose Galois group is isomorphic to \(S_5\). Prove that this is the case.
\item
This problem has three parts.
\begin{examsubparts}
\item[\textnormal{(a)}]
Prove that there exists a unitary ring homomorphism \(\phi:\mathbb{Z}[[X]]\otimes_{\mathbb{Z}}\mathbb{Q}\to\mathbb{Q}[[X]]\) such that \(\phi(f(X)\otimes 1)=f(X)\) for all \(f(X)\in\mathbb{Z}[[X]]\).
\item[\textnormal{(b)}]
Prove that~\(\phi\) is one-to-one.
\item[\textnormal{(c)}]
Prove that~\(\phi\) is not onto.
\end{examsubparts}
\item
Prove that an abelian group is divisible if and only if it is an injective~\(\mathbb{Z}\)-module.
\item
Let~\(R\) be a commutative ring with 1, let~\(M\) be a finitely generated~\(R\)-module, and let~\(I\) be an ideal of~\(R\) which is contained in every maximal ideal of~\(R\). Prove that if \(IM=M\) then \(M=0\).
\item
Let~\(R\) be a commutative Noetherian ring with 1. Let \(0=Q_1\cap\cdots\cap Q_n\) be a reduced primary decomposition of the zero ideal in~\(R\) and set \(P_i=\sqrt{Q_i}\) for \(1\leq i\leq n\). Prove that \(P_1\cup\cdots\cup P_n\) is the set of zero divisors of~\(R\).
\item
Let~\(R\) be a UFD. Prove that~\(R\) is integrally closed in its field of fractions.
\item
This problem has two parts.
\begin{examsubparts}
\item[\textnormal{(a)}]
Let~\(R\) be a ring. Prove that the Jacobson radical \(J(R)\) contains no nonzero idempotents.
\item[\textnormal{(b)}]
Give an example of a ring~\(R\) with a nonzero idempotent~\(e\) such that~\(e\) is a left quasi-regular element of~\(R\).
\end{examsubparts}
\item
Let~\(K\) be a field and let~\(A\) be a central simple algebra over~\(K\) such that \(\dim_K(A)=n\) is finite. Prove that \(A\otimes_K A^{\mathrm{op}}\cong M_{n\times n}(K)\).
\end{examproblems}
